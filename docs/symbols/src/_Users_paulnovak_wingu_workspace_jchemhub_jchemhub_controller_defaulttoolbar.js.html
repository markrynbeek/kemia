100 /**
101  * Initializes the given font menu button by adding default fonts to the menu.
102  * If jchemhub.controller.DefaultToolbar.setLocale was called to specify a
103  * locale for which locale-specific default fonts exist, those are added before
104  * common fonts.
105  * 
106  * @param {!goog.ui.Select}
107  *            button Font menu button.
108  */
109 jchemhub.controller.DefaultToolbar.addDefaultFonts = function(button) {
110 	// Normalize locale to lowercase, with a hyphen (see bug 1036165).
111 	var locale = jchemhub.controller.DefaultToolbar.locale_.replace(/_/, '-')
112 			.toLowerCase();
113 	// Add locale-specific default fonts, if any.
114 	var fontlist = [];
115 
116 	if (locale in jchemhub.controller.DefaultToolbar.I18N_FONTS_) {
117 		fontlist = jchemhub.controller.DefaultToolbar.I18N_FONTS_[locale];
118 	}
119 	if (fontlist.length) {
120 		jchemhub.controller.ToolbarFactory.addFonts(button, fontlist);
121 	}
122 	// Add locale-independent default fonts.
123 	jchemhub.controller.ToolbarFactory.addFonts(button,
124 			jchemhub.controller.DefaultToolbar.FONTS_);
125 };

247 /**
248  * Creates a {@link goog.ui.Toolbar} containing the specified set of toolbar
249  * buttons, and renders it into the given parent element. Each item in the
250  * {@code items} array must either be a {@link goog.editor.Command} (to create a
251  * built-in button) or a subclass of {@link goog.ui.Control} (to create a custom
252  * control).
253  * 
254  * @param {!Array.
255  *            <string|goog.ui.Control>} items Toolbar items; each must be a
256  *            {@link goog.editor.Command} or a {@link goog.ui.Control}.
257  * @param {!Element}
258  *            elem Toolbar parent element.
259  * @param {boolean=}
260  *            opt_isRightToLeft Whether the editor chrome is right-to-left;
261  *            defaults to the directionality of the toolbar parent element.
262  * @return {!goog.ui.Toolbar} Editor toolbar, rendered into the given parent
263  *         element.
264  */
265 jchemhub.controller.DefaultToolbar.makeToolbar = function(items, elem) {
266 	var domHelper = goog.dom.getDomHelper(elem);
267 	var controls = [];
268 
269 	for ( var i = 0, button; button = items[i]; i++) {
270 		if (goog.isString(button)) {
271 			button = jchemhub.controller.DefaultToolbar
272 					.makeBuiltInToolbarButton(button, domHelper);
273 		}
274 		if (button) {
275 			controls.push(button);
276 		}
277 	}
278 
279 	return jchemhub.controller.ToolbarFactory.makeToolbar(controls, elem);
280 };

282 /**
283  * Creates an instance of a subclass of {@link goog.ui.Button} for the given
284  * {@link goog.editor.Command}, or null if no built-in button exists for the
285  * command. Note that this function is only intended to create built-in buttons;
286  * please don't try to hack it!
287  * 
288  * @param {string}
289  *            command Editor command ID.
290  * @param {goog.dom.DomHelper=}
291  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
292  *            current document if unspecified.
293  * @return {goog.ui.Button} Toolbar button (null if no built-in button exists
294  *         for the command).
295  */
296 jchemhub.controller.DefaultToolbar.makeBuiltInToolbarButton = function(command,
297 		opt_domHelper) {
298 	var button;
299 	var descriptor = jchemhub.controller.DefaultToolbar.buttons_[command];
300 	if (descriptor) {
301 		// Default the factory method to makeToggleButton, since most built-in
302 		// toolbar buttons are toggle buttons. See also
303 		// jchemhub.controller.DefaultToolbar.BUTTONS_.
304 		var factory = descriptor.factory
305 				|| jchemhub.controller.ToolbarFactory.makeToggleButton;
306 		var id = descriptor.command;
307 		var tooltip = descriptor.tooltip;
308 		var caption = descriptor.caption;
309 		var classNames = descriptor.classes;
310 		// Default the DOM helper to the one for the current document.
311 		var domHelper = opt_domHelper || goog.dom.getDomHelper();
312 		// Instantiate the button based on the descriptor.
313 		button = factory(id, tooltip, caption, classNames, null, domHelper);
314 		// If this button's state should be queried when updating the toolbar,
315 		// set the button object's queryable property to true.
316 		if (descriptor.queryable) {
317 			button.queryable = true;
318 		}
319 	}
320 	return button;
321 };

342  *

344  *            id Button ID; must equal a {@link jchemhub.controller.Command} for
345  *            built-in buttons, anything else for custom buttons.
346  * @param {string}
347  *            tooltip Tooltip to be shown on hover.
348  * @param {goog.ui.ControlContent}
349  *            caption Button caption.
350  * @param {string=}
351  *            opt_classNames CSS class name(s) to apply to the caption element.
352  * @param {goog.ui.ButtonRenderer=}
353  *            opt_renderer Button renderer; defaults to
354  *            {@link goog.ui.ToolbarButtonRenderer} if unspecified.
355  * @param {goog.dom.DomHelper=}
356  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
357  *            current document if unspecified.
358  * @return {!goog.ui.Button} A toolbar button.
359  * @private
360  */
361 jchemhub.controller.DefaultToolbar.rtlButtonFactory_ = function(id, tooltip,
362 		caption, opt_classNames, opt_renderer, opt_domHelper) {
363 	var button = jchemhub.controller.ToolbarFactory.makeToggleButton(id,
364 			tooltip, caption, opt_classNames, opt_renderer, opt_domHelper);
365 	button.updateFromValue = function(value) {
366 		// Enable/disable right-to-left text editing mode in the toolbar.
367 		var isRtl = !!value;
368 		// Enable/disable a marker class on the toolbar's root element; the rest
369 		// is
370 		// done using CSS scoping in editortoolbar.css. This changes
371 		// direction-senitive toolbar icons (like indent/outdent)
372 		goog.dom.classes.enable(button.getParent().getElement(), goog
373 				.getCssName('tr-rtl-mode'), isRtl);
374 		button.setChecked(isRtl);
375 	}
376 	return button;
377 };

413  * Creates a toolbar button with the given ID, tooltip, and caption. Applies any
414  * custom CSS class names to the button's caption element. Used to create a font
415  * face button, filled with default fonts.
416  * 
417  * @param {string}
418  *            id Button ID; must equal a {@link jchemhub.controller.Command} for
419  *            built-in buttons, anything else for custom buttons.
420  * @param {string}
421  *            tooltip Tooltip to be shown on hover.
422  * @param {goog.ui.ControlContent}
423  *            caption Button caption.
424  * @param {string=}
425  *            opt_classNames CSS class name(s) to apply to the caption element.
426  * @param {goog.ui.MenuButtonRenderer=}
427  *            opt_renderer Button renderer; defaults to
428  *            {@link goog.ui.ToolbarMenuButtonRenderer} if unspecified.
429  * @param {goog.dom.DomHelper=}
430  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
431  *            current document if unspecified.
432  * @return {!goog.ui.Button} A toolbar button.
433  * @private
434  */
435 jchemhub.controller.DefaultToolbar.fontFaceFactory_ = function(id, tooltip,
436 		caption, opt_classNames, opt_renderer, opt_domHelper) {
437 	var button = jchemhub.controller.ToolbarFactory.makeSelectButton(id,
438 			tooltip, caption, opt_classNames, opt_renderer, opt_domHelper);
439 	jchemhub.controller.DefaultToolbar.addDefaultFonts(button);
440 	button
441 			.setDefaultCaption(jchemhub.controller.DefaultToolbar.MSG_FONT_NORMAL);
442 	// Font options don't have keyboard accelerators.
443 	goog.dom.classes.add(button.getMenu().getContentElement(), goog
444 			.getCssName('goog-menu-noaccel'));
445 
446 	// How to update this button's state.
447 	button.updateFromValue = function(value) {
448 		// Normalize value to null or a non-empty string (sometimes we get
449 		// the empty string, sometimes we get false...), extract the substring
450 		// up to the first comma to get the primary font name, and normalize
451 		// to lowercase. This allows us to map a font spec like "Arial,
452 		// Helvetica, sans-serif" to a font menu item.
453 		// TODO (attila): Try to make this more robust.
454 		var item = null;
455 		if (value && value.length > 0) {
456 			item = /** @type {goog.ui.MenuItem} */
457 			(button.getMenu().getChild(jchemhub.controller.ToolbarFactory
458 					.getPrimaryFont(value)));
459 		}
460 		var selectedItem = button.getSelectedItem();
461 		if (item != selectedItem) {
462 			button.setSelectedItem(item);
463 		}
464 	}
465 	return button;
466 };

482  * @param {goog.ui.MenuButtonRenderer=}
483  *            opt_renderer Button renderer; defaults to
484  *            {@link goog.ui.ToolbarMebuButtonRenderer} if unspecified.
485  * @param {goog.dom.DomHelper=}
486  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
487  *            current document if unspecified.
488  * @return {!goog.ui.Button} A toolbar button.
489  * @private

491 jchemhub.controller.DefaultToolbar.fontSizeFactory_ = function(id, tooltip,
492 		caption, opt_classNames, opt_renderer, opt_domHelper) {

494 			tooltip, caption, opt_classNames, opt_renderer, opt_domHelper);

498 	// Font size options don't have keyboard accelerators.
499 	goog.dom.classes.add(button.getMenu().getContentElement(), goog
500 			.getCssName('goog-menu-noaccel'));
501 	// How to update this button's state.
502 	button.updateFromValue = function(value) {
503 		// Webkit returns a string like '32px' instead of the equivalent
504 		// integer, so normalize that first.
505 		// NOTE: Gecko returns "6" so can't just normalize all
506 		// strings, only ones ending in "px".
507 		if (goog.isString(value) && goog.style.getLengthUnits(value) == 'px') {
508 			value = jchemhub.controller.ToolbarFactory
509 					.getLegacySizeFromPx(parseInt(value, 10));
510 		}
511 		// Normalize value to null or a positive integer (sometimes we get
512 		// the empty string, sometimes we get false, or -1 if the above
513 		// normalization didn't match to a particular 0-7 size)
514 		value = value > 0 ? value : null;
515 		if (value != button.getValue()) {
516 			button.setValue(value);
517 		}
518 	}
519 	return button;
520 };

522 /**
523  * Function to update the state of a color menu button.
524  * 
525  * @param {goog.ui.ToolbarColorMenuButton}
526  *            button The button to which the color menu is attached.
527  * @param {number}
528  *            value Color value to update to.
529  * @private
530  */
531 jchemhub.controller.DefaultToolbar.colorUpdateFromValue_ = function(button,
532 		value) {
533 	/** @preserveTry */
534 	try {
535 		if (goog.userAgent.IE) {
536 			// IE returns a number that, converted to hex, is a BGR color.
537 			// Convert from decimal to BGR to RGB.
538 			var hex = '000000' + value.toString(16);
539 			var bgr = hex.substr(hex.length - 6, 6);
540 			value = new goog.string.StringBuffer('#', bgr.substring(4, 6), bgr
541 					.substring(2, 4), bgr.substring(0, 2)).toString();
542 		}
543 		if (value != button.getValue()) {
544 			button.setValue(/** @type {string} */
545 			(value));
546 		}
547 	} catch (ex) {
548 		// TODO: Find out when/why this happens.
549 	}
550 };

620 /**
621  * Creates a toolbar button with the given ID, tooltip, and caption. Applies any
622  * custom CSS class names to the button's caption element. Use to create the
623  * format menu, prefilled with default formats.
624  * 
625  * @param {string}
626  *            id Button ID; must equal a {@link jchemhub.controller.Command} for
627  *            built-in buttons, anything else for custom buttons.
628  * @param {string}
629  *            tooltip Tooltip to be shown on hover.
630  * @param {goog.ui.ControlContent}
631  *            caption Button caption.
632  * @param {string=}
633  *            opt_classNames CSS class name(s) to apply to the caption element.
634  * @param {goog.ui.MenuButtonRenderer=}
635  *            opt_renderer Button renderer; defaults to
636  *            {@link goog.ui.ToolbarMenuButtonRenderer} if unspecified.
637  * @param {goog.dom.DomHelper=}
638  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
639  *            current document if unspecified.
640  * @return {!goog.ui.Button} A toolbar button.
641  * @private
642  */
643 jchemhub.controller.DefaultToolbar.formatBlockFactory_ = function(id, tooltip,
644 		caption, opt_classNames, opt_renderer, opt_domHelper) {
645 	var button = jchemhub.controller.ToolbarFactory.makeSelectButton(id,
646 			tooltip, caption, opt_classNames, opt_renderer, opt_domHelper);
647 	jchemhub.controller.DefaultToolbar.addDefaultFormatOptions(button);
648 	button
649 			.setDefaultCaption(jchemhub.controller.DefaultToolbar.MSG_FORMAT_NORMAL);
650 	// Format options don't have keyboard accelerators.
651 	goog.dom.classes.add(button.getMenu().getContentElement(), goog
652 			.getCssName('goog-menu-noaccel'));
653 	// How to update this button.
654 	button.updateFromValue = function(value) {
655 		// Normalize value to null or a nonempty string (sometimes we get
656 		// the empty string, sometimes we get false...)
657 		value = value && value.length > 0 ? value : null;
658 		if (value != button.getValue()) {
659 			button.setValue(value);
660 		}
661 	}
662 	return button;
663 };

786  * <li>{@code tooltip} - Tooltip text (optional); if unspecified, the button
787  * has no hover text
788  * <li>{@code caption} - Caption to display on the button (optional); if
789  * unspecified, the button has no text caption
790  * <li>{@code classes} - CSS class name(s) to be applied to the button's
791  * element when rendered (optional); if unspecified, defaults to 'tr-icon' plus
792  * 'tr-' followed by the command ID, but without any leading '+' character (e.g.
793  * if the command ID is '+undo', then {@code classes} defaults to 'tr-icon
794  * tr-undo')
795  * <li>{@code factory} - factory function used to create the button, which must
796  * accept {@code id}, {@code tooltip}, {@code caption}, and {@code classes}
797  * as arguments, and must return an instance of {@link goog.ui.Button} or an
798  * appropriate subclass (optional); if unspecified, defaults to
799  * {@link jchemhub.controller.DefaultToolbar.makeToggleButton}, since most
800  * built-in toolbar buttons are toggle buttons
801  * <li>(@code queryable} - Whether the button's state should be queried when
802  * updating the toolbar (optional).
803  * </ul>
804  * Note that this object is only used for creating toolbar buttons for built-in
805  * editor commands; custom buttons aren't listed here. Please don't try to hack
806  * this!
