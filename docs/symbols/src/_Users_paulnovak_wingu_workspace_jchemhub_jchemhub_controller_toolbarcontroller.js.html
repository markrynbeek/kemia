  9 /**
 10  * A class for managing the editor toolbar.  Acts as a bridge between
 11  * a {@link jchemhub.controller.ReactionEditor} and a {@link goog.ui.Toolbar}.
 12  *
 13  * The {@code toolbar} argument must be an instance of {@link goog.ui.Toolbar}
 14  * or a subclass.  This class doesn't care how the toolbar was created.  As
 15  * long as one or more controls hosted  in the toolbar have IDs that match
 16  * built-in {@link jchemhub.controller.ReactionEditor.Command}s, they will function as expected.  It is
 17  * the caller's responsibility to ensure that the toolbar is already rendered
 18  * or that it decorates an existing element.
 19  *
 20  *
 21  * @param {!jchemhub.controller.ReactionEditor} editor to be controlled by the
 22  *     toolbar.
 23  * @param {!goog.ui.Toolbar} toolbar Toolbar to control the editable editor.
 24  * @constructor
 25  * @extends {goog.events.EventTarget}
 26  */
 27 jchemhub.controller.ToolbarController = function(editor, toolbar) {
 28   goog.events.EventTarget.call(this);
 29 
 30   /**
 31    * Event handler to listen for editor events and user actions.
 32    * @type {!goog.events.EventHandler}
 33    * @private
 34    */
 35   this.handler_ = new goog.events.EventHandler(this);
 36 
 37   /**
 38    * The editor instance controlled by the toolbar.
 39    * @type {!jchemhub.controller.ToolbarController}
 40    * @private
 41    */
 42   this.editor_ = editor;
 43 
 44   /**
 45    * The toolbar that controls the editor.
 46    * @type {!goog.ui.Toolbar}
 47    * @private
 48    */
 49   this.toolbar_ = toolbar;
 50 
 51   /**
 52    * Editing commands whose state is to be queried when updating the toolbar.
 53    * @type {!Array.<string>}
 54    * @private
 55    */
 56   this.queryCommands_ = [];
 57 
 58   // Iterate over all buttons, and find those which correspond to
 59   // queryable commands. Add them to the list of commands to query on
 60   // each COMMAND_VALUE_CHANGE event.
 61   this.toolbar_.forEachChild(function(button) {
 62     if (button.queryable) {
 63       this.queryCommands_.push(this.getComponentId(button.getId()));
 64     }
 65   }, this);
 66 
 67   // Make sure the toolbar doesn't steal keyboard focus.
 68   this.toolbar_.setFocusable(false);
 69 
 70   // Hook up handlers that update the toolbar in response to editor events,
 71   // and to execute editor commands in response to toolbar events.
 72   this.handler_.
 73       listen(this.editor_, jchemhub.controller.ReactionEditor.EventType.COMMAND_VALUE_CHANGE,
 74           this.updateToolbar).
 75       listen(this.toolbar_, goog.ui.Component.EventType.ACTION,
 76           this.handleAction);
 77 };
 78 goog.inherits(jchemhub.controller.ToolbarController, goog.events.EventTarget);

 92   // The default implementation assumes that the component ID is the same as
 93   // the command constant.

183  * highlighted item, and closing the currently open menu (if any).
184  */
185 jchemhub.controller.ToolbarController.prototype.blur = function() {
186   // We can't just call this.toolbar_.getElement().blur(), because the toolbar
187   // element itself isn't focusable, so goog.ui.Container#handleBlur isn't
188   // registered to handle blur events.
189   this.toolbar_.handleBlur(null);

193 /** @inheritDoc */
194 jchemhub.controller.ToolbarController.prototype.disposeInternal = function() {
195   jchemhub.controller.ToolbarController.superClass_.disposeInternal.call(this);
196   if (this.handler_) {
197     this.handler_.dispose();
198     delete this.handler_;
199   }
200   if (this.toolbar_) {
201     this.toolbar_.dispose();
202     delete this.toolbar_;
203   }
204   delete this.editor_;
205   delete this.queryCommands_;
206 };

209 /**
210  * Updates the toolbar in response to editor events.  Specifically, updates
211  * button states based on {@code COMMAND_VALUE_CHANGE} events, reflecting the
212  * effective formatting of the selection.
213  * @param {goog.events.Event} e Editor event to handle.
214  * @protected
215  */
216 jchemhub.controller.ToolbarController.prototype.updateToolbar = function(e) {
217   if (!this.toolbar_.isEnabled() ||
218       !this.dispatchEvent(goog.ui.Component.EventType.CHANGE)) {
219     return;
220   }
221 
222   var state;
223 
224   /** @preserveTry */
225   try {
226     /** @type {Array.<string>} */
227     e.commands; // Added by dispatchEvent.
228 
229     // If the COMMAND_VALUE_CHANGE event specifies which commands changed
230     // state, then we only need to update those ones, otherwise update all
231     // commands.
232     state = /** @type {Object} */ (
233         this.editor_.queryCommandValue(e.commands || this.queryCommands_));
234   } catch (ex) {
235     // TODO: Find out when/why this happens.
236     state = {};
237   }
238 
239   this.updateToolbarFromState(state);
240 };
241 
242 
243 /**
244  * Updates the toolbar to reflect a given state.
245  * @param {Object} state Object mapping editor commands to values.
246  */
247 jchemhub.controller.ToolbarController.prototype.updateToolbarFromState =
248     function(state) {
249   for (var command in state) {
250     var button = this.toolbar_.getChild(this.getComponentId(command));
251     if (button) {
252       var value = state[command];
253       if (button.updateFromValue) {
254         button.updateFromValue(value);
255       } else {
256         button.setChecked(!!value);
257       }
258     }
259   }
260 };

263 /**
264  * Handles {@code ACTION} events dispatched by toolbar buttons in response to
265  * user actions by executing the corresponding editor command.
266  * @param {goog.events.Event} e Action event to handle.
267  * @protected
268  */
269 jchemhub.controller.ToolbarController.prototype.handleAction = function(e) {
270   var command = this.getCommand(e.target.getId());
271   this.editor_.execCommand(command, e.target.getValue());
272 };
