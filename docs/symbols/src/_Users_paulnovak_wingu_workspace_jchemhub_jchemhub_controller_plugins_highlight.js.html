 34 jchemhub.controller.plugins.Highlight.prototype.handleAtomMouseOver = function(e) {
 35 	if (!e.currentTarget.atomHighlightGroup) {
 36 		e.currentTarget.atomHighlightGroup = this.highlightAtom(e.atom);
 37 	} else {
 38 		e.currentTarget.atomHighlightGroup = this.highlightAtom(e.atom, e.currentTarget.atomHighlightGroup);
 39 	}
 40 
 41 };
 42 
 43 jchemhub.controller.plugins.Highlight.prototype.handleAtomMouseOut = function(e) {
 44 
 45 	if (e.currentTarget.atomHighlightGroup) {
 46 		e.currentTarget.atomHighlightGroup.clear();
 47 		e.currentTarget.atomHighlightGroup = undefined;
 48 	}
 49 
 50 };
 51 
 52 jchemhub.controller.plugins.Highlight.prototype.handleBondMouseOver = function(e) {
 53 	this.logger.info("handleBondMouseOver");
 54 	if (!e.currentTarget.bondHighlightGroup) {
 55 		e.currentTarget.bondHighlightGroup = this.highlightBond(e.bond);
 56 	} else {
 57 		e.currentTarget.bondHighlightGroup = this.highlightBond(e.bond, e.currentTarget.bondHighlightGroup);
 58 	}
 59 
 60 };
 61 jchemhub.controller.plugins.Highlight.prototype.handleBondMouseOut = function(e) {
 62 
 63 	if (e.currentTarget.bondHighlightGroup) {
 64 		e.currentTarget.bondHighlightGroup.clear();
 65 		e.currentTarget.bondHighlightGroup = undefined;
 66 	}
 67 
 68 };

 70 jchemhub.controller.plugins.Highlight.prototype.highlightBond = function(bond, opt_group){
 71 	return this.editorObject.reactionRenderer.moleculeRenderer.bondRendererFactory.get(bond).highlightOn(bond, opt_group);
 72 };
 73 
 74 jchemhub.controller.plugins.Highlight.prototype.highlightAtom = function(atom, opt_group){
 75 	return this.editorObject.reactionRenderer.moleculeRenderer.atomRenderer.highlightOn(atom, opt_group);
 76 };
