 10 /**
 11  * Class to render a reaction object to a graphics object
 12  * 
 13  * @constructor
 14  * @param parentEventTarget
 15  *            {goog.events.EventTarget}
 16  * @param graphics
 17  *            {goog.graphics.AbstractGraphics} graphics to draw on.
 18  * @extends {jchemhub.view.Renderer}
 19  */
 20 jchemhub.view.ReactionRenderer = function(controller, graphics, opt_config) {
 21 	jchemhub.view.Renderer.call(this, controller, graphics, opt_config,
 22 			jchemhub.view.ReactionRenderer.defaultConfig);
 23 	this.scale_factor = 1;
 24 	this.moleculeController = new jchemhub.controller.MoleculeController(
 25 			controller);
 26 	this.moleculeRenderer = new jchemhub.view.MoleculeRenderer(
 27 			this.moleculeController, graphics, this.config);
 28 	this.arrowRenderer = new jchemhub.view.ArrowRenderer(controller, graphics,
 29 			this.config);
 30 	this.plusRenderer = new jchemhub.view.PlusRenderer(controller, graphics,
 31 			this.config);
 32 }
 33 goog.inherits(jchemhub.view.ReactionRenderer, jchemhub.view.Renderer);
 34 /**
 35  * 
 36  * @param {jchemhub.model.Reaction}
 37  *            reaction
 38  * @return {goog.graphics.GroupElement}
 39  */
 40 jchemhub.view.ReactionRenderer.prototype.render = function(reaction) {
 41 	var previousReactant;
 42 	var transform = this.getTransform(reaction);
 43 	var group = this.graphics.createGroup();
 44 	goog.array.forEach(reaction.reactants, function(reactant) {
 45 		if (previousReactant) {
 46 			var center = this.center( [ previousReactant, reactant ]);
 47 			this.plusRenderer.render(center, transform, group);
 48 		}
 49 		previousReactant = reactant;
 50 		this.moleculeRenderer.render(reactant, transform, group);
 51 	}, this);
 52 
 53 	var reaction_center = this.center(goog.array.concat(reaction.reactants,
 54 			reaction.products));
 55 	this.arrowRenderer.render(reaction_center, transform, group);
 56 
 57 	var previousProduct = null;
 58 	goog.array.forEach(reaction.products, function(product) {
 59 		if (previousProduct) {
 60 			var center = this.center( [ previousProduct, product ]);
 61 			this.plusRenderer.render(center, transform, group);
 62 		}
 63 		previousProduct = product;
 64 		this.moleculeRenderer.render(product, transform, group);
 65 	}, this);
 66 	return group;
 67 }

 69 /**
 70  * finds center of an array of molecules
 71  * 
 72  * @return goog.math.Coordinate
 73  */
 74 jchemhub.view.ReactionRenderer.prototype.center = function(molecules) {
 75 	var bbox = this.boundingBox(molecules);
 76 	return new goog.math.Coordinate((bbox.left + bbox.right) / 2,
 77 			(bbox.top + bbox.bottom) / 2);
 78 }

 80 /**
 81  * finds bounding box of an array of molecules
 82  * 
 83  * @param molecules
 84  * @return goog.math.Box
 85  */
 86 
 87 jchemhub.view.ReactionRenderer.prototype.boundingBox = function(molecules) {
 88 	var atoms = goog.array.flatten(goog.array.map(molecules, function(mol) {
 89 		return mol.atoms;
 90 	}));
 91 	var coords = goog.array.map(atoms, function(a) {
 92 		return a.coord;
 93 	})
 94 	var m = this.config.get("margin");
 95 	return goog.math.Box.boundingBox.apply(null, coords).expand(m, m, m, m);
 96 }

107 /**
108  * 
109  * @param {Array.
110  *            <jchemhub.model.Molecule>} molecules
111  * @return {goog.math.Rect}
112  */
113 jchemhub.view.ReactionRenderer.prototype.boundingRect = function(molecules) {
114 	return goog.math.Rect.createFromBox(this.boundingBox(molecules));
115 }
116 /**
117  * 
118  * @param {jchemhub.model.Reaction}
119  *            reaction
120  * @return {jchemhub.graphics.AffineTransform}
121  */
122 jchemhub.view.ReactionRenderer.prototype.getTransform = function(reaction) {
123 	var molecules = goog.array.concat(reaction.reactants, reaction.products);
124 	var fromRect = this.boundingRect(molecules);
125 
126 	var toSize = fromRect.getSize().scaleToFit(this.graphics.getSize());
127 	var scale = this.scale_factor * toSize.width / fromRect.getSize().width;
128 
129 	var transform = new jchemhub.graphics.AffineTransform(scale, 0, 0, -scale,
130 			-fromRect.left * scale, -fromRect.top * scale);
131 
132 	return transform;
133 };
