 23 /**
 24  * Takes a font spec (e.g. "Arial, Helvetica, sans-serif") and returns the
 25  * primary font name, normalized to lowercase (e.g. "arial").
 26  * 
 27  * @param {string}
 28  *            fontSpec Font specification.
 29  * @return {string} The primary font name, in lowercase.
 30  */
 31 jchemhub.controller.ToolbarFactory.getPrimaryFont = function(fontSpec) {
 32 	var i = fontSpec.indexOf(',');
 33 	var fontName = (i != -1 ? fontSpec.substring(0, i) : fontSpec)
 34 			.toLowerCase();
 35 	// Strip leading/trailing quotes from the font name (bug 1050118).
 36 	return goog.string.stripQuotes(fontName, '"\'');
 37 };

 40  * Bulk-adds fonts to the given font menu button. The argument must be an array
 41  * of font descriptor objects, each of which must have the following attributes:
 42  * <ul>
 43  * <li>{@code caption} - Caption to show in the font menu (e.g. 'Tahoma')
 44  * <li>{@code value} - Value for the corresponding 'font-family' CSS style
 45  * (e.g. 'Tahoma, Arial, sans-serif')
 46  * </ul>
 47  * 
 48  * @param {!goog.ui.Select}
 49  *            button Font menu button.
 50  * @param {!Array.
 51  *            <{caption: string, value: string}>} fonts Array of font
 52  *            descriptors.
 53  */
 54 jchemhub.controller.ToolbarFactory.addFonts = function(button, fonts) {
 55 	goog.array.forEach(fonts, function(font) {
 56 		jchemhub.controller.ToolbarFactory.addFont(button, font.caption,
 57 				font.value);
 58 	});
 59 };

 61 /**
 62  * Adds a menu item to the given font menu button. The first font listed in the
 63  * {@code value} argument is considered the font ID, so adding two items whose
 64  * CSS style starts with the same font may lead to unpredictable results.
 65  * 
 66  * @param {!goog.ui.Select}
 67  *            button Font menu button.
 68  * @param {string}
 69  *            caption Caption to show for the font menu.
 70  * @param {string}
 71  *            value Value for the corresponding 'font-family' CSS style.
 72  */
 73 jchemhub.controller.ToolbarFactory.addFont = function(button, caption, value) {
 74 	// The font ID is the first font listed in the CSS style, normalized to
 75 	// lowercase.
 76 	var id = jchemhub.controller.ToolbarFactory.getPrimaryFont(value);
 77 
 78 	// Construct the option, and add it to the button.
 79 	var option = new goog.ui.Option(caption, value, button.dom_);
 80 	option.setId(id);
 81 	button.addItem(option);
 82 
 83 	// Captions are shown in their own font.
 84 	option.getContentElement().style.fontFamily = value;
 85 };

109 /**
110  * Adds a menu item to the given font size menu button. The {@code value}
111  * argument must be a legacy HTML font size in the 0-7 range.
112  * 
113  * @param {!goog.ui.Select}
114  *            button Font size menu button.
115  * @param {string}
116  *            caption Caption to show in the font size menu.
117  * @param {number}
118  *            value Value for the corresponding HTML font size.
119  */
120 jchemhub.controller.ToolbarFactory.addFontSize = function(button, caption,
121 		value) {
122 	// Construct the option, and add it to the button.
123 	var option = new goog.ui.Option(caption, value, button.dom_);
124 	option.setId(caption);
125 	button.addItem(option);
126 
127 	// Adjust the font size of the menu item and the height of the checkbox
128 	// element after they've been rendered by addItem(). Captions are shown in
129 	// the corresponding font size, and lining up the checkbox is tricky.
130 	var content = option.getContentElement();
131 	content.style.fontSize = jchemhub.controller.ToolbarFactory
132 			.getPxFromLegacySize(value) + 'px';
133 	content.firstChild.style.height = '1.1em';
134 };

136 /**
137  * Converts a legacy font size specification into an equivalent pixel size. For
138  * example, {@code <font size="6">} is {@code font-size: 32px;}, etc.
139  * 
140  * @param {number}
141  *            fontSize Legacy font size spec in the 0-7 range.
142  * @return {number} Equivalent pixel size.
143  */
144 jchemhub.controller.ToolbarFactory.getPxFromLegacySize = function(fontSize) {
145 	return jchemhub.controller.ToolbarFactory.LEGACY_SIZE_TO_PX_MAP_[fontSize] || 10;
146 };

148 /**
149  * Converts a pixel font size specification into an equivalent legacy size. For
150  * example, {@code font-size: 32px;} is {@code <font size="6">}, etc. If
151  * the given pixel size doesn't exactly match one of the legacy sizes, -1 is
152  * returned.
153  * 
154  * @param {number}
155  *            px Pixel font size.
156  * @return {number} Equivalent legacy size spec in the 0-7 range, or -1 if none
157  *         exists.
158  */
159 jchemhub.controller.ToolbarFactory.getLegacySizeFromPx = function(px) {
160 	// Use lastIndexOf to get the largest legacy size matching the pixel size
161 	// (most notably returning 1 instead of 0 for 10px).
162 	return goog.array.lastIndexOf(
163 			jchemhub.controller.ToolbarFactory.LEGACY_SIZE_TO_PX_MAP_, px);
164 };

166 /**
167  * Map of legacy font sizes (0-7) to equivalent pixel sizes.
168  * 
169  * @type {Array.<number>}
170  * @private
171  */
172 jchemhub.controller.ToolbarFactory.LEGACY_SIZE_TO_PX_MAP_ = [ 10, 10, 13, 16,
173 		18, 24, 32, 48 ];

217 /**
218  * Creates a {@link goog.ui.Toolbar} containing the specified set of toolbar
219  * buttons, and renders it into the given parent element. Each item in the
220  * {@code items} array must a {@link goog.ui.Control}.
221  * 
222  * @param {!Array.
223  *            <goog.ui.Control>} items Toolbar items; each must be a
224  *            {@link goog.ui.Control}.
225  * @param {!Element}
226  *            elem Toolbar parent element.
227  * @param {boolean=}
228  *            opt_isRightToLeft Whether the editor chrome is right-to-left;
229  *            defaults to the directionality of the toolbar parent element.
230  * @return {!goog.ui.Toolbar} Editor toolbar, rendered into the given parent
231  *         element.
232  */
233 jchemhub.controller.ToolbarFactory.makeToolbar = function(items, elem,
234 		opt_isRightToLeft) {
235 	var domHelper = goog.dom.getDomHelper(elem);
236 
237 	// Create an empty horizontal toolbar using the default renderer.
238 	var toolbar = new goog.ui.Toolbar(goog.ui.ToolbarRenderer.getInstance(),
239 			goog.ui.Container.Orientation.HORIZONTAL, domHelper);
240 
241 	// Optimization: Explicitly test for the directionality of the parent
242 	// element here, so we can set it for both the toolbar and its children,
243 	// saving a lot of expensive calls to goog.style.isRightToLeft() during
244 	// rendering.
245 	var isRightToLeft = opt_isRightToLeft || goog.style.isRightToLeft(elem);
246 	toolbar.setRightToLeft(isRightToLeft);
247 
248 	// Optimization: Set the toolbar to non-focusable before it is rendered,
249 	// to avoid creating unnecessary keyboard event handler objects.
250 	toolbar.setFocusable(false);
251 
252 	for ( var i = 0, button; button = items[i]; i++) {
253 		// Optimization: Set the button to non-focusable before it is rendered,
254 		// to avoid creating unnecessary keyboard event handler objects. Also
255 		// set
256 		// the directionality of the button explicitly, to avoid expensive calls
257 		// to goog.style.isRightToLeft() during rendering.
258 		button.setSupportedState(goog.ui.Component.State.FOCUSED, false);
259 		button.setRightToLeft(isRightToLeft);
260 		toolbar.addChild(button, true);
261 	}
262 
263 	toolbar.render(elem);
264 	return toolbar;
265 };

270  *

291 			.createContent_(caption, opt_classNames, opt_domHelper),
292 			opt_renderer, opt_domHelper);
293 	button.setId(id);
294 	button.setTooltip(tooltip);
295 	return button;

362 /**
363  * Creates a select button with the given ID, tooltip, and caption. Applies any
364  * custom CSS class names to the button's root element. The button returned
365  * doesn't have an actual menu attached; use {@link goog.ui.Select#setMenu} to
366  * attach a {@link goog.ui.Menu} containing {@link goog.ui.Option}s to the
367  * select button.
368  * 
369  * @param {string}
370  *            id Button ID; must equal a {@link goog.editor.Command} for
371  *            built-in buttons, anything else for custom buttons.
372  * @param {string}
373  *            tooltip Tooltip to be shown on hover.
374  * @param {goog.ui.ControlContent}
375  *            caption Button caption; used as the default caption when nothing
376  *            is selected.
377  * @param {string=}
378  *            opt_classNames CSS class name(s) to apply to the button's root
379  *            element.
380  * @param {goog.ui.MenuButtonRenderer=}
381  *            opt_renderer Button renderer; defaults to
382  *            {@link goog.ui.ToolbarMenuButtonRenderer} if unspecified.
383  * @param {goog.dom.DomHelper=}
384  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
385  *            current document if unspecified.
386  * @return {!goog.ui.Select} A select button.
387  */
388 jchemhub.controller.ToolbarFactory.makeSelectButton = function(id, tooltip,
389 		caption, opt_classNames, opt_renderer, opt_domHelper) {
390 	var button = new goog.ui.ToolbarSelect(null, null, opt_renderer,
391 			opt_domHelper);
392 	if (opt_classNames) {
393 		// Unlike the other button types, for goog.ui.Select buttons we apply
394 		// the
395 		// extra class names to the root element, because for select buttons the
396 		// caption isn't stable (as it changes each time the selection changes).
397 		goog.array.forEach(opt_classNames.split(/\s+/), button.addClassName,
398 				button);
399 	}
400 	button.addClassName(goog.getCssName('goog-toolbar-select'));
401 	button.setDefaultCaption(caption);
402 	button.setId(id);
403 	button.setTooltip(tooltip);
404 	return button;
405 };

441 /**
442  * Creates a new DIV that wraps a button caption, optionally applying CSS class
443  * names to it. Used as a helper function in button factory methods.
444  * 
445  * @param {goog.ui.ControlContent}
446  *            caption Button caption.
447  * @param {string=}
448  *            opt_classNames CSS class name(s) to apply to the DIV that wraps
449  *            the caption (if any).
450  * @param {goog.dom.DomHelper=}
451  *            opt_domHelper DOM helper, used for DOM creation; defaults to the
452  *            current document if unspecified.
453  * @return {!Element} DIV that wraps the caption.
454  * @private
455  */
456 jchemhub.controller.ToolbarFactory.createContent_ = function(caption,
457 		opt_classNames, opt_domHelper) {
458 	// FF2 doesn't like empty DIVs, especially when rendered right-to-left.
459 	if ((!caption || caption == '') && goog.userAgent.GECKO
460 			&& !goog.userAgent.isVersion('1.9a')) {
461 		caption = goog.string.Unicode.NBSP;
462 	}
463 	return (opt_domHelper || goog.dom.getDomHelper()).createDom(
464 			goog.dom.TagName.DIV, opt_classNames ? {
465 				'class' : opt_classNames
466 			} : null, caption);
467 };
