1 goog.provide('jchemhub.view.BondRendererFactory');
  2 goog.require('jchemhub.view.SingleBondRenderer');
  3 goog.require('jchemhub.view.DoubleBondRenderer');
  4 goog.require('jchemhub.view.TripleBondRenderer');
  5 goog.require('jchemhub.view.QuadrupleBondRenderer');
  6 goog.require('jchemhub.view.SingleUpBondRenderer');
  7 goog.require('jchemhub.view.SingleDownBondRenderer');
  8 goog.require('jchemhub.view.SingleUpOrDownBondRenderer');
  9 
 10 /**
 11  * factory class for BondRenderers
 12  * 
 13  * @param {goog.graphics.AbstractGraphics} graphics
 14  * @param {Object} opt_config
 15  * @constructor
 16  */
 17 jchemhub.view.BondRendererFactory = function(controller, graphics, opt_config) {
 18 	this.controller = controller;
 19 	this.graphics = graphics;
 20 	this.config = new goog.structs.Map();
 21 	if (opt_config) {
 22 		this.config.addAll(opt_config); // merge optional config
 23 	}
 24 }
 25 
 26 jchemhub.view.BondRendererFactory.prototype.get = function(bond) {
 27 	if (bond instanceof jchemhub.model.SingleBondUp) {
 28 		if (!this.singleUpBondRenderer) {
 29 			this.singleUpBondRenderer = new jchemhub.view.SingleUpBondRenderer(
 30 					this.controller, this.graphics, this.config);
 31 		}
 32 		return this.singleUpBondRenderer;
 33 	}
 34 	if (bond instanceof jchemhub.model.SingleBondDown) {
 35 		if (!this.singleDownBondRenderer) {
 36 			this.singleDownBondRenderer = new jchemhub.view.SingleDownBondRenderer(
 37 					this.controller, this.graphics, this.config);
 38 		}
 39 		return this.singleDownBondRenderer;
 40 	}
 41 	if (bond instanceof jchemhub.model.SingleBondUpOrDown) {
 42 		if (!this.singleUpOrDownBondRenderer) {
 43 			this.singleUpOrDownBondRenderer = new jchemhub.view.SingleUpOrDownBondRenderer(
 44 					this.controller, this.graphics, this.config);
 45 		}
 46 		return this.singleUpOrDownBondRenderer;
 47 	}
 48 	if (bond instanceof jchemhub.model.SingleBond) {
 49 		if (!this.singleBondRenderer) {
 50 			this.singleBondRenderer = new jchemhub.view.SingleBondRenderer(
 51 					this.controller, this.graphics, this.config);
 52 		}
 53 		return this.singleBondRenderer;
 54 	}
 55 	if (bond instanceof jchemhub.model.DoubleBond) {
 56 		if (!this.doubleBondRenderer) {
 57 			this.doubleBondRenderer = new jchemhub.view.DoubleBondRenderer(
 58 					this.controller, this.graphics, this.config);
 59 		}
 60 		return this.doubleBondRenderer;
 61 	}
 62 	if (bond instanceof jchemhub.model.TripleBond) {
 63 		if (!this.tripleBondRenderer) {
 64 			this.tripleBondRenderer = new jchemhub.view.TripleBondRenderer(
 65 					this.controller, this.graphics, this.config);
 66 		}
 67 		return this.tripleBondRenderer;
 68 	}
 69 	if (bond instanceof jchemhub.model.QuadrupleBond) {
 70 		if (!this.quadrupleBondRenderer) {
 71 			this.quadrupleBondRenderer = new jchemhub.view.QuadrupleBondRenderer(
 72 					this.controller, this.graphics, this.config);
 73 		}
 74 		return this.quadrupleBondRenderer;
 75 	}
 76 };
