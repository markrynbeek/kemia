 19 jchemhub.view.SingleBondRenderer.prototype.render = function(bond, transform,
 20 		group) {
 21 	jchemhub.view.SingleBondRenderer.superClass_.render.call(this, bond,
 22 			transform, group);
 23 
 24 	var bondPath = new goog.graphics.Path();
 25 	var bondStroke = new goog.graphics.Stroke(
 26 			this.config.get("bond").stroke.width,
 27 			this.config.get("bond").stroke.color);
 28 	var bondFill = null;
 29 
 30 	var coords = transform.transformCoords( [ bond.source.coord,
 31 			bond.target.coord ]);
 32 
 33 	bondPath.moveTo(coords[0].x, coords[0].y);
 34 	bondPath.lineTo(coords[1].x, coords[1].y);
 35 
 36 	this.graphics.drawPath(bondPath, bondStroke, bondFill, group);

 38 }
