  6 /**
  7  * Class to render an atom object to a graphics object
  8  * 
  9  * @constructor
 10  * @param graphics
 11  *            {goog.graphics.AbstractGraphics} graphics to draw on.
 12  * @extends {jchemhub.view.Renderer}
 13  */
 14 jchemhub.view.AtomRenderer = function(controller, graphics, opt_config) {
 15 	jchemhub.view.Renderer.call(this, controller, graphics, opt_config,
 16 			jchemhub.view.AtomRenderer.defaultConfig);
 17 }
 18 goog.inherits(jchemhub.view.AtomRenderer, jchemhub.view.Renderer);
 19 /**
 20  * 
 21  * @param {jchemhub.model.Atom}
 22  *            atom
 23  * @param {jchemhub.graphics.AffineTransform}
 24  *            transform
 25  * @return {goog.graphics.GroupElement}
 26  */
 27 jchemhub.view.AtomRenderer.prototype.render = function(atom, transform) {
 28 	this.transform = transform;
 29 
 30 	var atom_config = this.config.get("atom");
 31 	var color = this.config.get(atom.symbol) ? this.config.get(atom.symbol).color
 32 			: atom_config.color;
 33 
 34 	var scale = transform.getScaleX();
 35 
 36 	var font = new goog.graphics.Font(scale / 1.8, atom_config.fontName);
 37 	var stroke = new goog.graphics.Stroke(atom_config.stroke.width, color);
 38 	var fill = new goog.graphics.SolidFill(color);
 39 
 40 	var point = transform.transformCoords( [ atom.coord ])[0];
 41 	var symbol = this.compoundSymbol(atom);
 42 	var graphics = this.graphics;
 43 	var w = symbol.text.length * 0.55 * font.size;
 44 	var h = font.size;
 45 	var group = graphics.createGroup();
 46 
 47 	if (symbol.text) {
 48 		graphics
 49 				.drawEllipse(point.x, point.y, h * 0.7, h * 0.7, null,
 50 						new goog.graphics.SolidFill(this.config
 51 								.get("background").color), group);
 52 
 53 		graphics.drawText(symbol.text, point.x - w / 2, point.y - h / 2, w, h,
 54 				symbol.justification, null, font, stroke, fill, group);
 55 		if (symbol.justification == 'left') {
 56 			if (symbol.subscript || symbol.superscript) {
 57 				var subSize = this.config.get("subscriptSize");
 58 				if (symbol.subscript) {
 59 					graphics.drawText(symbol.subscript, point.x + w * 0.9,
 60 							point.y, subSize, subSize, 'center', null, font,
 61 							stroke, fill, group);
 62 				}
 63 				if (symbol.superscript) {
 64 					graphics.drawText(symbol.superscript, point.x + w, point.y
 65 							- h * 0.8, subSize, subSize, 'center', null, font,
 66 							stroke, fill, group);
 67 				}
 68 			}
 69 		} else if (symbol.justification == 'right') {
 70 			if (symbol.subscript || symbol.superscript) {
 71 				var subSize = this.config.get("subscriptSize");
 72 				if (symbol.subscript) {
 73 					graphics.drawText('H', point.x - w * 3, point.y - h / 2, w,
 74 							h, 'center', null, font, stroke, fill, group);
 75 					graphics.drawText(symbol.subscript, point.x - w * 1.8,
 76 							point.y, subSize, subSize, 'center', null, font,
 77 							stroke, fill, group);
 78 				}
 79 				if (symbol.superscript) {
 80 					graphics.drawText(symbol.superscript, point.x + w, point.y
 81 							- h * 0.8, subSize, subSize, 'center', null, font,
 82 							stroke, fill, group);
 83 				}
 84 			}
 85 		}
 86 	} else {
 87 		graphics
 88 				.drawEllipse(point.x, point.y, h * 0.7, h * 0.7, null,
 89 						new goog.graphics.SolidFill(this.config
 90 								.get("background").color, 0.001), group);
 91 	}
 92 
 93 	group.addEventListener(goog.events.EventType.MOUSEOVER, goog.bind(
 94 			this.controller.handleMouseOver, this.controller, atom));
 95 	group.addEventListener(goog.events.EventType.MOUSEOUT, goog.bind(
 96 			this.controller.handleMouseOut, this.controller, atom));
 97 	group.addEventListener(goog.events.EventType.MOUSEDOWN, goog.bind(
 98 			this.controller.handleMouseDown, this.controller, atom));
 99 	return group;
100 
101 };

103 jchemhub.view.AtomRenderer.prototype.highlightOn = function(atom, opt_group) {
104 	var atom_config = this.config.get("atom");
105 	var strokeWidth = atom_config.stroke.width * 4;
106 	var color = this.config.get(atom.symbol) ? this.config.get(atom.symbol).color
107 			: atom_config.color;
108 	var stroke = new goog.graphics.Stroke(strokeWidth, color);
109 	var fill = null
110 	var radius = atom_config.highlight.radius * this.transform.getScaleX();
111 	var coords = this.transform.transformCoords( [ atom.coord ])[0];
112 
113 	if (!opt_group) {
114 		opt_group = this.graphics.createGroup();
115 	}
116 	this.graphics.drawCircle(coords.x, coords.y, radius, stroke, fill,
117 			opt_group);
118 
119 	return opt_group;
120 };

122 /**
123  * return a compound symbol (e.g. NH, CH3), the plain symbol, or ""
124  * 
125  * @param{jchemhun.model.Atom} atom
126  * @return {String}
127  */
128 jchemhub.view.AtomRenderer.prototype.compoundSymbol = function(atom) {
129 	var retval = {
130 		text : "",
131 		justification : 'center',
132 		superscript : '',
133 		subscript : ''
134 	};
135 	if (atom.symbol != "C" || atom.countBonds() == 1) {
136 		// terminal atom may need compound atom name
137 		var hydrogen_count = atom.hydrogenCount();
138 		if (hydrogen_count == 0) {
139 			retval.text = atom.symbol;
140 			// could have H on left, depending on bond slope
141 		} else {
142 			bond_direction = jchemhub.view.AtomRenderer.bondDirection(atom);
143 			var justification = 'center';
144 			if (bond_direction == "SW" || bond_direction == "W"
145 					|| bond_direction == "NW") {
146 				justification = 'right';
147 				if (hydrogen_count == 1)
148 					retval.text = 'H';
149 				retval.text += atom.symbol;
150 			} else {
151 				justification = 'left';
152 				retval.text = atom.symbol + 'H';
153 			}
154 			if (hydrogen_count > 1)
155 				retval.subscript = String(hydrogen_count);
156 		}
157 		if (atom.charge) {
158 			if (atom.charge > 1) {
159 				retval.superscript += '+' + atom.charge;
160 			} else if (atom.charge < -1) {
161 				retval.superscript += atom.charge;
162 			} else if (atom.charge == -1) {
163 				retval.superscript = '-';
164 			} else if (atom.charge == 1) {
165 				retval.superscript = '+';
166 			}
167 		}
168 
169 	} else {
170 		retval.text = "";
171 	}
172 	retval.justification = justification;
173 	return retval;
174 };

176 /**
177  * return an angle between 0 and 360 at which the i-th bond to this atom is
178  * drawn
179  * 
180  * @param{integer} i-th bond to this atom
181  * @return number
182  */
183 jchemhub.view.AtomRenderer.bondOrientation = function(atom, i) {
184 	var bond = atom.bonds.getValues()[i];
185 	var target = bond.target.coord;
186 	var source = bond.source.coord;
187 
188 	var dy = target.y - source.y;
189 	var dx = target.x - source.x;
190 	if (atom == bond.source) {
191 		dx = -dx;
192 		dy = -dy;
193 	}
194 	var angle = Math.atan2(dy, dx) * 180 / Math.PI;
195 	if (angle < 0)
196 		angle = 360 + angle; // now angle is 0 - 360
197 	return angle;
198 };
