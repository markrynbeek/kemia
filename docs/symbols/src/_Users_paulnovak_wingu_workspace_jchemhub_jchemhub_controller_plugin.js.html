  8 /**
  9  * Abstract API for reaction editor plugins.
 10  * 
 11  * @constructor
 12  * @extends {goog.events.EventTarget}
 13  */
 14 jchemhub.controller.Plugin = function() {
 15 	goog.events.EventTarget.call(this);
 16 
 17 	/**
 18 	 * Whether this plugin is enabled for the registered field object.
 19 	 * 
 20 	 * @type {boolean}
 21 	 * @private
 22 	 */
 23 	this.enabled_ = this.activeOnUneditableEditor();
 24 };
 25 goog.inherits(jchemhub.controller.Plugin, goog.events.EventTarget);
 26 
 27 /**
 28  * @return {boolean} If true, editor will not disable the command when the
 29  *         editor becomes uneditable.
 30  */
 31 jchemhub.controller.Plugin.prototype.activeOnUneditableEditor = goog.functions.FALSE;

 33 /**
 34  * Registers the reaction editor object for use with this plugin.
 35  * 
 36  * @param {jchemhub.controller.ReactionEditor}
 37  *            fieldObject The reaction editor object.
 38  */
 39 jchemhub.controller.Plugin.prototype.registerEditorObject = function(editorObject) {
 40 	this.editorObject = editorObject;
 41 };
 42 
 43 /**
 44  * Enables this plugin for the specified, registered reaction editor object. A
 45  * reaction editor object should only be enabled when it is loaded.
 46  * 
 47  * @param {jchemhub.controller.ReactionEditor}
 48  *            editorObject The field object.
 49  */
 50 jchemhub.controller.Plugin.prototype.enable = function(editorObject) {
 51 	if (this.editorObject == editorObject) {
 52 		this.enabled_ = true;
 53 	} else {
 54 		this.logger
 55 				.severe('Trying to enable an unregistered field with ' + 'this plugin.');
 56 	}
 57 };
 58 
 59 /**
 60  * Disables this plugin for the specified, registered reaction editor object.
 61  * 
 62  * @param {jchemhub.controller.ReactionEditor}
 63  *            editorObject The reaction editor object.
 64  */
 65 jchemhub.controller.Plugin.prototype.disable = function(editorObject) {
 66 	if (this.editorObject == editorObject) {
 67 		this.enabled_ = false;
 68 	} else {
 69 		this.logger
 70 				.severe('Trying to disable an unregistered field ' + 'with this plugin.');
 71 	}
 72 };

 74 /**
 75  * The logger for this plugin.

 77  * @type {goog.debug.Logger}
 78  * @protected
 79  */
 80 jchemhub.controller.Plugin.prototype.logger = goog.debug.Logger
 81 		.getLogger('jchemhub.controller.Plugin');

 83 /**
 84  * Returns whether this plugin is enabled for the reaction editor object.
 85  * 
 86  * @param {jchemhub.controller.ReactionEditor}
 87  *            editorObject The reaction editor object.
 88  * @return {boolean} Whether this plugin is enabled for the reaction editor
 89  *         object.
 90  */
 91 jchemhub.controller.Plugin.prototype.isEnabled = function(editorObject) {
 92 	return this.editorObject == editorObject ? this.enabled_ : false;
 93 };
 94 
 95 /** @inheritDoc */
 96 jchemhub.controller.Plugin.prototype.disposeInternal = function() {
 97 	if (this.editorObject) {
 98 		this.unregisterEditorObject(this.editorObject);
 99 	}
100 
101 	jchemhub.controller.Plugin.superClass_.disposeInternal.call(this);
102 };
103 
104 /**
105  * Unregisters and disables this plugin for the current editor object.
106  * 
107  */
108 jchemhub.controller.Plugin.prototype.unregisterEditorObject = function() {
109 	if (this.editorObject) {
110 		this.disable(this.editorObject);
111 		this.editorObject = null;
112 	}
113 };

200 /**
201  * Handles execCommand. This default implementation handles dispatching
202  * BEFORECHANGE, CHANGE, and SELECTIONCHANGE events, and calls
203  * execCommandInternal to perform the actual command. Plugins that want to do
204  * their own event dispatching should override execCommand, otherwise it is
205  * preferred to only override execCommandInternal.
206  * 
207  * This version of execCommand will only work for single field plugins.
208  * Multi-field plugins must override execCommand.
209  * 
210  * @param {string}
211  *            command The command to execute.
212  * @param {...*}
213  *            var_args Any additional parameters needed to execute the command.
214  * @return {*} The result of the execCommand, if any.
215  */
216 jchemhub.controller.Plugin.prototype.execCommand = function(command, var_args) {
217 	// TODO: Replace all uses of isSilentCommand with plugins that just
218 	// override this base execCommand method.
219 	var silent = this.isSilentCommand(command);
220 	if (!silent) {
221 		// Stop listening to mutation events in Firefox while text formatting
222 		// is happening. This prevents us from trying to size the field in the
223 		// middle of an execCommand, catching the field in a strange
224 		// intermediary
225 		// state where both replacement nodes and original nodes are appended to
226 		// the dom. Note that change events get turned back on by
227 		// editorObject.dispatchChange.
228 		if (goog.userAgent.GECKO) {
229 			this.editorObject.stopChangeEvents(true, true);
230 		}
231 
232 		this.editorObject.dispatchBeforeChange();
233 	}
234 
235 	try {
236 		var result = this.execCommandInternal.apply(this, arguments);
237 	} finally {
238 		// If the above execCommandInternal call throws an exception, we still
239 		// need
240 		// to turn change events back on (see http://b/issue?id=1471355).
241 		// NOTE: If if you add to or change the methods called in this finally
242 		// block, please add them as expected calls to the unit test function
243 		// testExecCommandException().
244 		if (!silent) {
245 			// dispatchChange includes a call to startChangeEvents, which
246 			// unwinds the
247 			// call to stopChangeEvents made before the try block.
248 			this.editorObject.dispatchChange();
249 
250 			this.editorObject.dispatchSelectionChangeEvent();
251 
252 		}
253 	}
254 
255 	return result;
256 };
