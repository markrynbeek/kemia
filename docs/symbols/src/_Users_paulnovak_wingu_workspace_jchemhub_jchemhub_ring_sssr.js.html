 52     /**
 53      * Make a deep copy of an array
 54      */
 55     function deepCopy(arr) {
 56         var newArray = [];
 57         for (var i = 0, li = arr.length; i < li; i++) {
 58             var item = arr[i];
 59             if (item instanceof Array) {
 60                 newArray.push(deepCopy(item));
 61             } else {
 62                 newArray.push(item);
 63             }
 64         }
 65         return newArray;
 66     }

 83     /**
 84      * Create a n x n matrix with all elements set to 0.
 85      */
 86     function createEmptyMatrix(n) {
 87         var matrix = [];
 88         for (var i = 0; i < n; i++) {
 89             var row = [];
 90             for (var j = 0; j < n; j++) {
 91                 row.push(0);
 92             }
 93             matrix.push(row);
 94         }
 95         return matrix;
 96     }

 98     /**
 99      * Create an initial distance matrix. This is  the D matrix from the paper.
100      */
101     function createWeightMatrix(molecule, n) {
102         var matrix = [];
103         for (var i = 0; i < n; i++) {
104             var row = [];
105             for (var j = 0; j < n; j++) {
106                 if (i == j) {
107                     row.push(0);
108                 } else if (molecule.findBond(molecule.getAtom(i), molecule.getAtom(j))) {
109                     row.push(1);
110                 } else {
111                     row.push(99999999);
112                 }
113             }
114             matrix.push(row);
115         }
116         return matrix;
117     }

119     /**
120      * Create an empty Path-Included Distance matrix. This is an n x n matrix 
121      * with all elements set to an empty list (i.e. there is no path between 
122      * i and j)
123      */
124     function createEmptyPIDMatrix(n) {
125         var matrix = [];
126         for (var i = 0; i < n; i++) {
127             var row = [];
128             for (var j = 0; j < n; j++) {
129                 row.push([]);
130             }
131             matrix.push(row);
132         }
133         return matrix;
134 
135     }

137     /**
138      * Create the initial Pe matrix. The supplementary information 
139      * indicates this should be an empty PID matrix, this makes no sense though.
140      * We initialize it with adding all bonds as paths of length 1 when atom i 
141      * and j are connected.
142      */
143     function createPIDMatrix(molecule, n) {
144         var matrix = [];
145         for (var i = 0; i < n; i++) {
146             var row = [];
147             for (var j = 0; j < n; j++) {
148                 var beginAtom = molecule.getAtom(i);
149                 var endAtom = molecule.getAtom(j);
150                 var bond = molecule.findBond(beginAtom, endAtom);
151                 if (bond) {
152                     row.push([[molecule.indexOfBond(bond)]]);
153                 } else {
154                     row.push([]);
155                 }
156             }
157             matrix.push(row);
158         }
159         return matrix;
160 
161     }

163     /**
164      * Update a path included distance matrix element (i.e. lhs) by merging 
165      * p1 with p2. 
166      */
167     function appendPath(lhs, p1, p2) {
168         if (!lhs.length) {
169             lhs[0] = p1[0].concat(p2[0]);
170         } else {
171             lhs.push(p1[0].concat(p2[0]));
172         }
173     }

176     /**
177      * Create the two Path-Included Distance matrices (Pe and Pe') and the 
178      * distance matrix D. This is Algorithm 1 in the supplementary information
179      * that comes with the paper.
180      */
181     function makePIDMatrixes(molecule) {
182         var n = molecule.countAtoms();
183         var D = createWeightMatrix(molecule, n);
184         var Pe1 = createPIDMatrix(molecule, n); // Pe
185         var Pe2 = createEmptyPIDMatrix(n); // Pe'
186         var lastD = D;
187 
188         //debug("Pe =<br>" + matrixToHTML(Pe1));
189         //debug("Pe' =<br>" + matrixToHTML(Pe2));
190 
191         for (var k = 0; k < n; k++) {
192             for (var i = 0; i < n; i++) {
193                 for (var j = 0; j < n; j++) {
194                     var lastPathLength = lastD[i][j];
195                     var pathLength = lastD[i][k] + lastD[k][j];
196                     var path1 = Pe1[i][k];
197                     var path2 = Pe1[k][j];
198 
199                     // ignore invalid paths
200                     if (pathLength == 100000000) { continue; }
201 
202                     if (lastPathLength > pathLength) {
203                         if (lastPathLength == pathLength + 1) {
204                             // a new shortest path = previous shortest path -1 => Pe' <- Pe
205                             Pe2[i][j] = deepCopy(Pe1[i][j]);
206                         } else {
207                             Pe2[i][j] = [];
208                         }
209 
210                         // a new shortest path is found
211                         D[i][j] = pathLength;
212                         Pe1[i][j] = [ path1[0].concat(path2[0]) ]; // change path
213                     } else if (lastPathLength == pathLength) {
214                         // another shortest path is found
215                         if (path1.length && path2.length) {
216                             appendPath(Pe1[i][j], path1, path2); // append path
217                         }
218                     } else if (lastPathLength == pathLength - 1) {
219                         // shortest path + 1 found
220                         appendPath(Pe2[i][j], path1, path2); // append path
221                     } else {
222                         D[i][j] = lastD[i][j];
223                     }
224                 }
225             }
226         }
