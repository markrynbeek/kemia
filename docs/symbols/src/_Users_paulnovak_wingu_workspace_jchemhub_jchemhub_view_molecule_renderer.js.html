1 goog.provide('jchemhub.view.MoleculeRenderer');
  2 goog.require('jchemhub.controller.BondController');
  3 goog.require('jchemhub.view.BondRenderer');
  4 goog.require('jchemhub.view.BondRendererFactory');
  5 goog.require('jchemhub.view.AtomRenderer');
  6 goog.require('jchemhub.controller.AtomController');
  7 
  8 /**
  9  * Class to render a molecule object to a graphics object
 10  * 
 11  * @constructor
 12  * @param graphics
 13  *            {goog.graphics.AbstractGraphics} graphics to draw on.
 14  * @extends {jchemhub.view.Renderer}
 15  */
 16 jchemhub.view.MoleculeRenderer = function(controller, graphics, opt_config) {
 17 	jchemhub.view.Renderer.call(this, controller, graphics, opt_config, jchemhub.view.MoleculeRenderer.defaultConfig);
 18 	this.bondController = new jchemhub.controller.BondController(controller);
 19 	this.bondRendererFactory = new jchemhub.view.BondRendererFactory(this.bondController, graphics);
 20 	this.atomController = new jchemhub.controller.AtomController(controller);
 21 	this.atomRenderer = new jchemhub.view.AtomRenderer(this.atomController, graphics);
 22 }
 23 goog.inherits(jchemhub.view.MoleculeRenderer, jchemhub.view.Renderer);
 24 
 25 jchemhub.view.MoleculeRenderer.prototype.render = function(molecule, trans, group) {
 26 	this.transform = trans;
 27 	goog.array.forEach(molecule.bonds, function(bond) {
 28 		this.bondRendererFactory.get(bond).render(bond, trans, group);
 29 	}, this);
 30 	goog.array.forEach(molecule.atoms, function(atom) {
 31 		this.atomRenderer.render(atom, trans, this.atomController);
 32 	}, this);
 33 }
