  4 /**
  5  * Class to render a quadruple bond object to a graphics object

  8  * @param graphics
  9  *            {goog.graphics.AbstractGraphics} graphics to draw on.

 12 jchemhub.view.QuadrupleBondRenderer = function(controller, graphics, opt_config) {
 13 	jchemhub.view.BondRenderer.call(this, controller, graphics, opt_config,
 14 			jchemhub.view.QuadrupleBondRenderer.defaultConfig);

 16 }
 17 goog.inherits(jchemhub.view.QuadrupleBondRenderer, jchemhub.view.BondRenderer);
 18 
 19 jchemhub.view.QuadrupleBondRenderer.prototype.render = function(bond,
 20 		transform, group) {
 21 	jchemhub.view.QuadrupleBondRenderer.superClass_.render.call(this, bond,
 22 			transform, group);
 23 	var strokeWidth = this.config.get("bond").stroke.width;
 24 	var bondStroke = new goog.graphics.Stroke(strokeWidth, this.config
 25 			.get("bond").stroke.color);
 26 	var bondFill = null;
 27 
 28 	var theta = jchemhub.view.BondRenderer.getTheta(bond);
 29 
 30 	var angle_left = theta + (Math.PI / 2);
 31 	var angle_right = theta - (Math.PI / 2);
 32 
 33 	var bondWidth = goog.math.Coordinate.distance(bond.source.coord,
 34 			bond.target.coord) / 6;
 35 	var transleft1 = new jchemhub.graphics.AffineTransform(1, 0, 0, 1, Math
 36 			.cos(angle_left)
 37 			* bondWidth / 3, Math.sin(angle_left) * bondWidth / 3);
 38 
 39 	var transright1 = new jchemhub.graphics.AffineTransform(1, 0, 0, 1, Math
 40 			.cos(angle_right)
 41 			* bondWidth / 3, Math.sin(angle_right) * bondWidth / 3);
 42 	var transleft2 = new jchemhub.graphics.AffineTransform(1, 0, 0, 1, Math
 43 			.cos(angle_left)
 44 			* bondWidth, Math.sin(angle_left) * bondWidth);
 45 
 46 	var transright2 = new jchemhub.graphics.AffineTransform(1, 0, 0, 1, Math
 47 			.cos(angle_right)
 48 			* bondWidth, Math.sin(angle_right) * bondWidth);
 49 
 50 	var leftside1 = transleft1.transformCoords( [ bond.source.coord,
 51 			bond.target.coord ]);
 52 	var rightside1 = transright1.transformCoords( [ bond.source.coord,
 53 			bond.target.coord ]);
 54 
 55 	var leftside2 = transleft2.transformCoords( [ bond.source.coord,
 56 			bond.target.coord ]);
 57 	var rightside2 = transright2.transformCoords( [ bond.source.coord,
 58 			bond.target.coord ]);
 59 
 60 	var coords = transform.transformCoords( [ leftside1[0], leftside1[1],
 61 			rightside1[0], rightside1[1], leftside2[0], leftside2[1],
 62 			rightside2[0], rightside2[1] ]);
 63 
 64 	var bondPath = new goog.graphics.Path();
 65 	bondPath.moveTo(coords[0].x, coords[0].y);
 66 	bondPath.lineTo(coords[1].x, coords[1].y);
 67 
 68 	bondPath.moveTo(coords[2].x, coords[2].y);
 69 	bondPath.lineTo(coords[3].x, coords[3].y);
 70 
 71 	 bondPath.moveTo(coords[4].x, coords[4].y);
 72 	 bondPath.lineTo(coords[5].x, coords[5].y);
 73 	 
 74 	 bondPath.moveTo(coords[6].x, coords[6].y);
 75 	 bondPath.lineTo(coords[7].x, coords[7].y);
 76 
 77 	this.graphics.drawPath(bondPath, bondStroke, bondFill, group);
 78 }
