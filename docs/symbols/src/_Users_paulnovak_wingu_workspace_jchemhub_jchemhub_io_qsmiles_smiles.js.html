 62 jchemhub.io.smiles.parse = function (smi) {
 63 	items = smi.match(this.smiPattern);
 64 	var mol = new jchemhub.model.Molecule(smi);
 65 	var natoms = 0;
 66 	var previous_atom;
 67 	var bond_type = jchemhub.io.smiles.BondType.NONE;
 68 	var bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
 69 	var branch = new Array();
 70 	var ring   = new Array();
 71 	var errstr = "";
 72 	for (var i in items) {
 73 		var item = items[i];
 74 		if (item == this.punctuation.nobond) {
 75 		} else if (item == this.punctuation.openbranch) {
 76 			branch.push(previous_atom);
 77 		} else if (item == this.punctuation.closebranch) {
 78 			if (branch.length) {
 79 				previous_atom = branch.pop();
 80 			} else {
 81 				errstr = " unbalanced parens";
 82 			}
 83 		} else if (item == this.punctuation.singlebond) {
 84 			bond_type = jchemhub.io.smiles.BondType.SINGLE_BOND;
 85 		} else if (item == this.punctuation.doublebond) {
 86 			bond_type = jchemhub.io.smiles.BondType.DOUBLE_BOND;
 87 		} else if (item == this.punctuation.triplebond) {
 88 			bond_type = jchemhub.io.smiles.BondType.TRIPLE_BOND;
 89 		} else if (item == this.punctuation.quadbond) {
 90 			bond_type = jchemhub.io.smiles.BondType.QUAD_BOND;
 91 		} else if (item == this.punctuation.aromaticbond) {
 92 			bond_type = jchemhub.io.smiles.BondType.AROMATIC_BOND
 93 		} else if (item[0] == this.punctuation.ringclosure) {
 94 			ringid = parseInt(item[1]+item[2]);
 95 			ring_atom = ring[ringid];
 96 			if (ring_atom) {
 97 				//var bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
 98 				mol.addBond(this.createBond(bond_type, bond_stereo, previous_atom, ring_atom));
 99 				bond_type = jchemhub.io.smiles.BondType.NONE;
100 				bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
101 				ring[ringid] = null;
102 			} else {
103 				ring[ringid] = previous_atom;
104 			}
105 		} else if (item == this.punctuation.cis) {
106 		} else if (item == this.punctuation.trans) {
107  		} else if (!isNaN(ringid = parseInt(item))) {
108 			ring_atom = ring[ringid];
109 			if (ring_atom) {
110 				//var bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
111 				mol.addBond(this.createBond(bond_type, bond_stereo, previous_atom, ring_atom));
112 				bond_type = jchemhub.io.smiles.BondType.NONE;
113 				bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
114 				ring[ringid] = null;
115 			} else {
116 				ring[ringid] = previous_atom;
117 			}
118 		} else {
119   			smi_atom = this.parseAtom(item);
120   			if (smi_atom.symbol) {
121 				natoms += 1;
122 				var atom = new jchemhub.model.Atom(smi_atom.symbol,0,0,smi_atom.charge,smi_atom.aromatic,smi_atom.isotope);
123 				if (previous_atom) {
124 					//var bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
125 					//if (smi_atom.stereo) bond_stereo = smi_atom.stereo;
126 					mol.addBond(this.createBond(bond_type, bond_stereo, previous_atom, atom));
127 					bond_type = jchemhub.io.smiles.BondType.NONE;
128 					bond_stereo = jchemhub.io.smiles.BondStereo.NONE;
129 				}
130 				mol.addAtom(atom);
131 				bond_stereo = smi_atom.stereo; // bond to next atom specified in this atom spec
132 				previous_atom = atom;
133 			} else {
134 				errstr =  " unknown atom " + item;
135 			}
136 		}
137 		if (errstr) {
138 			throw new Error(smi + errstr);
139 			break;
140 		}
141 	}
142 
143 	if (this.sanityCheck(branch, ring, bond_type, bond_stereo)) {
144 		return mol;
145 	} else {
146 		return null;
147 	}
148 };
149 
150 jchemhub.io.smiles.sanityCheck = function (branch, ring, bond_type, bond_stereo) {
151 	if (branch.length) {
152 		throw new Error(smi + " unbalanced parens");
153 		return false;
154 	}
155 	for (var i=0; i<ring.length; ++i) {
156 		if (ring[i]) {
157 			throw new Error(smi + " unclosed rings");
158 			return false;
159 		}
160 	}
161 	if (bond_type != this.BondType.NONE) {
162 		throw new Error(smi + " unpaired bond " + bond_type);
163 		return false;
164 	}
165 	/*
166 	if (bond_stereo != this.BondStereo.NONE) {
167 		throw new Error(smi + " unpaired stereo " + bond_stereo);
168 		return false;
169 	}
170 	*/
171 	return true;
172 };

174 jchemhub.io.smiles.parseAtom = function (item) {
175   var atom = {isotope:null, symbol:null, stereo:jchemhub.io.smiles.BondStereo.NONE, hcount:null, charge:null, aromatic:false};
176   var atomProp = this.atomPattern.exec(item);
177   if (atomProp) {
178     atom.isotope = atomProp[1];
179 
180     // periodicTable has entries for c,n,o,s,as,se
181     if (this.periodicTable[atomProp[2]]) atom.symbol = atomProp[2];
182     
183     if (atomProp[3] == jchemhub.io.smiles.BondStereo.CLOCKWISE) {
184 	    atom.stereo = atomProp[3];
185 	} else if (atomProp[3] == jchemhub.io.smiles.BondStereo.COUNTER_CLOCKWISE) {
186 	    atom.stereo = atomProp[3]; 
187 	} else {
188 		atom.stereo = jchemhub.io.smiles.BondStereo.NONE;
189 	}
190     
191     if (atomProp[4] == 'H') {
192       if (atomProp[5]) {
193         atom.hcount = atomProp[5];
194       } else {
195         atom.hcount = 1;
196       }
197     }
198     
199     if (atomProp[6] == "+") {
200       atom.charge = 1;
201     } else if (atomProp[6] == "-") {
202       atom.charge = -1;
203     } else {
204       atom.charge = parseInt(atomProp[6]);
205     }
206   
207     } else {
208     if (this.specialAtoms[item]) {
209       atom.symbol = item;
210     }
211   }
212   if (this.aromaticAtoms[atom.symbol]) {
213 	  atom.aromatic = true;
214 	  if (atom.symbol.length == 1) {
215 		  atom.symbol = atom.symbol.toUpperCase();
216 	  } else {
217 		  atom.symbol = atom.symbol[0].toUpperCase() + atom.symbol[1];
218 	  }
219   }
220   return atom;
221 };

223 /**
224  * factory method for bonds
225  *
226  * @param{jchemhub.io.smiles.BondType}type bond-type code
227  * @param{jchemhub.io.smiles.BondStereo}stereo stereo-type code
228  * @param{jchemhub.model.Atom} source atom at source end of bond
229  * @param{jchemhub.model.Atom} target atom at target end of bond
230  *
231  * @return{jchemhub.model.Bond}
232  */
233 jchemhub.io.smiles.createBond = function(type, stereo, source, target) {
234 		var atype = type;
235 		if (type == jchemhub.io.smiles.BondType.NONE) {
236 			if (source.aromatic && target.aromatic) {
237 				atype = jchemhub.io.smiles.BondType.AROMATIC_BOND;
238 			} else {
239 				atype = jchemhub.io.smiles.BondType.SINGLE_BOND;
240 			}
241 		}
242         switch (atype) {
243         
244         case jchemhub.io.smiles.BondType.SINGLE_BOND:
245                 switch (stereo) {
246                 case jchemhub.io.smiles.BondStereo.CLOCKWISE:
247                         return new jchemhub.model.SingleBondUp(source, target);
248                 case jchemhub.io.smiles.BondStereo.COUNTER_CLOCKWISE:
249                         return new jchemhub.model.SingleBondDown(source, target);
250                 case jchemhub.io.smiles.BondStereo.NONE:
251                         return new jchemhub.model.SingleBond(source, target);
252                 default:
253                         throw new Error("invalid bond type/stereo [" + type + "]/["
254                                         + stereo + "]");
255                 };
256         case jchemhub.io.smiles.BondType.DOUBLE_BOND:
257                 return new jchemhub.model.DoubleBond(source, target);
258         case jchemhub.io.smiles.BondType.TRIPLE_BOND:
259                 return new jchemhub.model.TripleBond(source, target);
260         case jchemhub.io.smiles.BondType.AROMATIC_BOND:
261                 return new jchemhub.model.AromaticBond(source, target);
262         case jchemhub.io.smiles.BondType.ANY:
263         default:
264                 throw new Error("invalid bond type/stereo [" + type + "]/[" + stereo
265                                 + "]");
266         };
267 };
