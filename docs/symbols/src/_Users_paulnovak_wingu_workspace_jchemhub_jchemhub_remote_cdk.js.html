  5 goog.provide('jchemhub.remote.CDK');
  6 goog.require('goog.net.Jsonp');
  7 
  8 jchemhub.remote.CDK=function(_url) {
  9     this.serverUrl = _url ? _url : "http://localhost:8080/";
 10 }

 12 /**
 13  * Sends the given payload to the URL specified at the construction
 14  * time. The reply is delivered to the given replyCallback. If the
 15  * errorCallback is specified and the reply does not arrive within the
 16  * timeout period set on this channel, the errorCallback is invoked
 17  * with the original payload.
 18  *
 19  * @param {Object}  params TODO write doc.
 20  *
 21  * @param {Function=} replyCallback A function expecting one
 22  *     argument, called when the reply arrives, with the response data.
 23  *
 24  * @param {Function=} errorCallback A function expecting one
 25  *     argument, called on timeout, with the payload.
 26  *
 27  * @return {Object} A request descriptor that may be used to cancel this
 28  *     transmission, or null, if the message may not be cancelled.
 29  */
 30 
 31 jchemhub.remote.CDK.prototype.smilesToMolfile=function(params,replyCallback,errorCallback){
 32     var jsonp = new goog.net.Jsonp(new goog.Uri(this.serverUrl+'parseSmiles'));
 33     return jsonp.send(params,replyCallback,errorCallback);
 34 };
