 15 goog.provide('jchemhub.query.SmilesCompiler');
 16 
 17 goog.require('jchemhub.io.smiles');
 18 goog.require('jchemhub.query.MoleculeCompiler');
 19 
 20 (function() {
 21 
 22     /**
 23      * The Smiles Query Compiler compiles SMILES strings into queries. This
 24      * class implements {@link jchemhub.query.IQueryCompiler}. <a href="../Substructure Search.html">Substructure Search</a>
 25      * @class Smiles Query Compiler
 26      * @see jchemhub.query.IQueryCompiler
 27      */
 28     jchemhub.query.SmilesCompiler = {};
 29 
 30     /**
 31      * Compile a query from smiles string.
 32      * @param smiles The smiles string
 33      * @return {jchemhub.query.IQuery}
 34      */
 35     jchemhub.query.SmilesCompiler.compile = function(/**string*/smiles) {
 36         var molecule = jchemhub.io.smiles.parse(smiles);
 37         var query = jchemhub.query.MoleculeCompiler.compile(molecule);
 38         return query;
 39     };
 40  
 41 }());
