1 goog.provide('jchemhub.view.BondRenderer');
  2 goog.require('jchemhub.view.Renderer');
  3 goog.require('jchemhub.math.Line');

  5 /**
  6  * Class to render a bond object to a graphics object
  7  * 
  8  * @constructor
  9  * @param graphics
 10  *            {goog.graphics.AbstractGraphics} graphics to draw on.
 11  * @extends {jchemhub.view.Renderer}
 12  */
 13 jchemhub.view.BondRenderer = function(controller, graphics, opt_config) {
 14 	jchemhub.view.Renderer.call(this, controller, graphics, opt_config,
 15 			jchemhub.view.BondRenderer.defaultConfig);
 16 }
 17 goog.inherits(jchemhub.view.BondRenderer, jchemhub.view.Renderer);
 18 /**
 19  * renders the box surrounding the bond to serve as a click target
 20  * 
 21  * @param {jchemhub.model.Bond} bond
 22  * @param {jchemhub.graphics.AffineTransform} transform
 23  * @return {goog.graphics.GroupElement}
 24  */
 25 jchemhub.view.BondRenderer.prototype.render = function(bond, transform) {
 26 	this.transform = transform;
 27 
 28 	var fill = new goog.graphics.SolidFill('red', 0.001); // 'transparent'
 29 	// fill
 30 	var stroke = null;
 31 	var radius = this.config.get("highlight").radius ;
 32 	var theta = jchemhub.view.BondRenderer.getTheta(bond);
 33 
 34 	var angle_left = theta + (Math.PI / 2);
 35 	var angle_right = theta - (Math.PI / 2);
 36 
 37 	var transleft = new jchemhub.graphics.AffineTransform(1, 0, 0, 1, Math
 38 			.cos(angle_left)
 39 			* radius, Math.sin(angle_left) * radius);
 40 
 41 	var transright = new jchemhub.graphics.AffineTransform(1, 0, 0, 1, Math
 42 			.cos(angle_right)
 43 			* radius, Math.sin(angle_right) * radius);
 44 
 45 	var leftside = transleft.transformCoords( [ bond.source.coord,
 46 			bond.target.coord ]);
 47 	var rightside = transright.transformCoords( [ bond.source.coord,
 48 			bond.target.coord ]);
 49 
 50 	var boxCoords = transform.transformCoords( [ leftside[0], leftside[1],
 51 			rightside[0], rightside[1] ]);
 52 
 53 	bondBoxPath = new goog.graphics.Path();
 54 	bondBoxPath.moveTo(boxCoords[0].x, boxCoords[0].y);
 55 	bondBoxPath.lineTo(boxCoords[2].x, boxCoords[2].y);
 56 	bondBoxPath.lineTo(boxCoords[3].x, boxCoords[3].y);
 57 	bondBoxPath.lineTo(boxCoords[1].x, boxCoords[1].y);
 58 	bondBoxPath.close();
 59 
 60 	var group = this.graphics.createGroup();
 61 	this.graphics.drawPath(bondBoxPath, stroke, fill, group);
 62 	group.addEventListener(goog.events.EventType.MOUSEOVER, goog.bind(
 63 			this.controller.handleMouseOver, this.controller, bond));
 64 	group.addEventListener(goog.events.EventType.MOUSEOUT, goog.bind(
 65 			this.controller.handleMouseOut, this.controller, bond));
 66 	group.addEventListener(goog.events.EventType.MOUSEDOWN, goog.bind(
 67 			this.controller.handleMouseDown, this.controller, bond));
 68 	
 69 	return group;
 70 
 71 };

 73 jchemhub.view.BondRenderer.prototype.highlightOn = function(bond, opt_group) {
 74 
 75 	var strokeWidth = this.config.get("bond").stroke.width * 2;
 76 	var stroke = new goog.graphics.Stroke(strokeWidth, this.config
 77 			.get("highlight").color);
 78 	var fill = null
 79 	var radius = this.config.get("highlight").radius
 80 			* this.transform.getScaleX();
 81 	var theta = -jchemhub.view.BondRenderer.getTheta(bond) * 180 / Math.PI;
 82 	var angle = theta + 90;
 83 
 84 	var arcExtent;
 85 	if (theta <= 0) {
 86 		arcExtent = (bond.source.coord.y <= bond.target.coord.y) ? 180 : -180;
 87 	} else {
 88 		arcExtent = (bond.source.coord.y > bond.target.coord.y) ? 180 : -180;
 89 	}
 90 
 91 	var coords = this.transform.transformCoords( [ bond.source.coord,
 92 			bond.target.coord ]);
 93 
 94 	var path = new goog.graphics.Path();
 95 	path.arc(coords[0].x, coords[0].y, radius, radius, angle, arcExtent);
 96 	path.arc(coords[1].x, coords[1].y, radius, radius, angle, -arcExtent);
 97 
 98 	if (!opt_group) {
 99 		opt_group = this.graphics.createGroup();
100 	}
101 	this.graphics.drawPath(path, stroke, fill, opt_group);
102 	return opt_group;
103 }

105 /**
106  * 
107  * @return{number} bond angle of elevation
108  */
109 jchemhub.view.BondRenderer.getTheta = function(bond) {
110 	return new jchemhub.math.Line(bond.source.coord, bond.target.coord)
111 			.getTheta();
112 }
