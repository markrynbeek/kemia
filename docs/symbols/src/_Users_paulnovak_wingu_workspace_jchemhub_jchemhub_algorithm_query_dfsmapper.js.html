 15 goog.provide('jchemhub.query.DFSMapper');

 17 goog.require('goog.structs.Map');

 19 (function() {
 20 
 21     function simpleSort(a, b) {
 22         return a - b;
 23     }
 24 
 25     jchemhub.query.DFSMapper = function(query) {
 26 
 27         var Type = { MapFirst: 0, MapUnique: 1, MapAll: 2 };
 28 
 29         this.query = query;
 30 
 31         /**
 32          * Select a start atom with the least number of neighbors. This ensures
 33          * there is a path to all atoms in the query.
 34          */
 35         function getStartAtom(query) {
 36             var startAtom = query.getAtom(0);
 37             var leastNbrs = startAtom.neighbors.length;
 38             for (var i = 0, li = query.countAtoms(); i < li; i++) {
 39                 var qatom = query.getAtom(i);
 40                 if (qatom.neighbors.length < leastNbrs) {
 41                     startAtom = qatom;
 42                     leastNbrs = qatom.neighbors.length;
 43                 }
 44             }
 45             return startAtom;
 46         }
 47 
 48         /**
 49          * State to represent the current mapped state
 50          */
 51         function State(type, query, queried) {
 52             this.type = type;
 53             this.query = query;
 54             this.queried = queried;
 55             this.queryPath = [];
 56             this.queriedPath = [];
 57             this.candidates = [];
 58         }
 59 
 60         /**
 61          * Check if the current state is a full mapping of the query.
 62          */
 63         function checkForMap(state, maps) {
 64             // store the mapping if all atoms are mapped
 65             if (state.queryPath.length === state.query.countAtoms()) {
 66                 // create the map
 67                 var map = new goog.structs.Map();
 68                 for (var k = 0, kl = state.queryPath.length; k < kl; k++) {
 69                     map.set(state.query.indexOfAtom(state.queryPath[k]), state.queried.indexOfAtom(state.queriedPath[k]));
 70                 }
 71                 if (state.type === Type.MapUnique) {
 72                     var values = map.getValues();
 73                     values.sort(simpleSort);
 74                     var isUnique = true;
 75                     for (k = 0, kl = maps.length; k < kl; k++) {
 76                         var kValues = maps[k].getValues();
 77                         kValues.sort(simpleSort);
 78                         if (goog.array.equals(values, kValues)) {
 79                             isUnique = false;
 80                         }
 81                     }
 82                     if (isUnique) {
 83                         maps.push(map);
 84                     }
 85                 } else {
 86                     maps.push(map);
 87                 }
 88 
 89                 if (state.type === Type.MapFirst) {
 90                     return;
 91                 }
 92             }
 93         }
 94 
 95         /**
 96          * Match the candidate atoms a bonds.
 97          */
 98         function matchCandidate(state, queryAtom, queriedAtom, queryNbr, queriedNbr, maps) {
 99             // make sure the neighbor atom isn't in the paths already
100             if (goog.array.indexOf(state.queryPath, queryNbr) !== -1) {
101                 return false;
102             }
103             if (goog.array.indexOf(state.queriedPath, queriedNbr) !== -1) {
104                 return false;
105             }
106 
107             // check if the atoms match
108             if (!queryNbr.matches(queriedNbr)) {
109                 return false;
110             }
111 
112             var queryBond = state.query.findBond(queryAtom, queryNbr);
113             var queriedBond = state.queried.findBond(queriedAtom, queriedNbr);
114 
115             // check if the bonds match
116             if (!queryBond.matches(queriedBond)) {
117                 return false;
118             }
119 
120             // add the neighbors to the paths
121             state.queryPath.push(queryNbr);
122             state.queriedPath.push(queriedNbr);
123 
124             // check if this is a full match
125             checkForMap(state, maps);
126 
127             return true;
128         }
129 
130         /**
131          * The depth-first isomorphism algorithm.
132          */
133         function DFS(state, queryAtom, queriedAtom, maps) {
134             var queryNbrs = queryAtom.getNeighbors();
135             var queriedNbrs = queriedAtom.getNeighbors();
136 
137             // load the possible candidates
138             var candidates = [];
139             for (var i = 0, li = queryNbrs.length; i < li; i++) {
140                 var queryNbr = queryNbrs[i];
141                 for (var j = 0, lj = queriedNbrs.length; j < lj; j++) {
142                     var queriedNbr = queriedNbrs[j];
143                     candidates.push({ queryAtom: queryAtom, queryNbr: queryNbr, queriedAtom: queriedAtom, queriedNbr: queriedNbr });
144                 }
145             }
146             // save the candidates for later (used to explore branches)
147             if (state.candidates.length) {
148                 state.candidates.push(state.candidates[state.candidates.length-1].concat(candidates));
149             } else {
150                 state.candidates.push(candidates);
151             }
152 
153             // do the mapping by checking the candidates
154             while (state.candidates[state.candidates.length-1].length) {
155                 var candidate = state.candidates[state.candidates.length-1].pop();
156                 if (matchCandidate(state, candidate.queryAtom, candidate.queriedAtom, candidate.queryNbr, candidate.queriedNbr, maps)) {
157                     DFS(state, candidate.queryNbr, candidate.queriedNbr, maps);
158 
159                     // backtrack
160                     state.queryPath.pop();
161                     state.queriedPath.pop();
162                     state.candidates.pop();
163                 }
164             }
165         }
166 
167         /**
168          * Helper function for mapAllCallback, mapUniqueCallback and MapFirstCallback
169          */
170         function mapNext(i, type, query, queryAtom, queried, maps, callback) {
171             var state = new State(type, query, queried);
172             var queriedAtom = queried.getAtom(i);
173 
174             if (queryAtom.matches(queriedAtom)) {
175                 state.queryPath.push(queryAtom);
176                 state.queriedPath.push(queriedAtom);
177                 DFS(state, queryAtom, queriedAtom, maps);
178             }
179 
180             i++;
181             if (i < queried.countAtoms()) {
182                 var nextBitOfWork = function() { mapNext(i, type, query, queryAtom, queried, maps, callback); };
183                 setTimeout(nextBitOfWork, 0);
184             } else {
185                 callback(maps);
186             }
187         }
188 
189         /**
190          * Get all mappings of the query on the queried molecule.
191          * @param {jchemhub.model.Molecule} queried The queried molecule.
192          * @return {Array.<goog.structs.Set>} The mappings
193          */
194         this.mapAll = function(queried) {
195             var maps = [];
196             var queryAtom = getStartAtom(this.query);
197             for (var i = 0, li = queried.countAtoms(); i < li; i++) {
198                 var state = new State(Type.MapAll, this.query, queried);
199                 var queriedAtom = queried.getAtom(i);
200                 if (!queryAtom.matches(queriedAtom)) {
201                     continue
202                 }
203 
204                 if (this.query.countAtoms() > 1) {
205                     state.queryPath.push(queryAtom);
206                     state.queriedPath.push(queriedAtom);
207                     DFS(state, queryAtom, queriedAtom, maps);
208                 } else {
209                     var map = new goog.structs.Map();
210                     map.set(state.query.indexOfAtom(queryAtom), state.queried.indexOfAtom(queriedAtom));
211                     maps.push(map);
212                 }
213             }
214 
215             return maps;
216         };
217 
218         /**
219          * Get all mappings of the query on the queried molecule. The specified
220          * callback function will be called with the found maps as argument.
221          * Unlike mapAll, this function regulary gives control back to the
222          * browser, preventing the GUI to lock up.
223          * @param {jchemhub.model.Molecule} queried The queried molecule.
224          * @param {Function} callback The callback function to report results.
225          */
226         this.mapAllCallback = function(queried, callback) {
227             var maps = [];
228             var queryAtom = getStartAtom(this.query);
229             var i = 0;
230             mapNext(i, Type.MapAll, this.query, queryAtom, queried, maps, callback);
231         };
232 
233         /**
234          * Get all unique mappings of the query on the queried molecule.
235          * @param {jchemhub.model.Molecule} queried The queried molecule.
236          * @return {Array.<goog.structs.Set>} The unique mappings
237          */
238         this.mapUnique = function(queried) {
239             var maps = [];
240             var queryAtom = getStartAtom(this.query);
241             for (var i = 0, li = queried.countAtoms(); i < li; i++) {
242                 var state = new State(Type.MapUnique, this.query, queried);
243                 var queriedAtom = queried.getAtom(i);
244                 if (!queryAtom.matches(queriedAtom)) {
245                     continue;
246                 }
247                 if (this.query.countAtoms() > 1) {
248                     state.queryPath.push(queryAtom);
249                     state.queriedPath.push(queriedAtom);
250                     DFS(state, queryAtom, queriedAtom, maps);
251                 } else {
252                     var map = new goog.structs.Map();
253                     map.set(state.query.indexOfAtom(queryAtom), state.queried.indexOfAtom(queriedAtom));
254                     maps.push(map);
255                 }
256             }
257 
258             return maps;
259         };

261         /**
262          * Get all unique mappings of the query on the queried molecule. The
263          * specified callback function will be called with the found maps as
264          * argument. Unlike mapUnique, this function regulary gives control
265          * back to the browser preventing the GUI to lock up.
266          * @param {jchemhub.model.Molecule} queried The queried molecule.
267          * @param {Function} callback The callback function to report results.
268          */
269         this.mapUniqueCallback = function(queried, callback) {
270             var maps = [];
271             var queryAtom = getStartAtom(this.query);
272             var i = 0;
273             mapNext(i, Type.MapUnique, this.query, queryAtom, queried, maps, callback);
274         };

277         /**
278          * Get the first mappings of the query on the queried molecule.
279          * @param {jchemhub.model.Molecule} queried The queried molecule.
280          * @return {goog.structs.Set} The mapping
281          *
