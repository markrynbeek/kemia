  8 /**
  9  * Class representing an atom
 10  * 
 11  * @param {string}
 12  *            symbol, Atom symbol
 13  * @param {number}
 14  *            x, X-coordinate of atom.
 15  * @param {number}
 16  *            y, Y-coordinate of atom.
 17  * @param {number}
 18  *            opt_charge, Charge of atom, defaults to 0.
 19  * @constructor
 20  */
 21 jchemhub.model.Atom=function(symbol, x, y, opt_charge, opt_aromatic, opt_isotope)
 22 {
 23 	/**
 24 	 * Atom symbol
 25 	 * 
 26 	 * @type {string}
 27 	 */
 28     this.symbol = symbol;
 29     
 30     /**
 31      * 2d coordinates 
 32      * @type{goog.math.Coordinate}
 33      */
 34     this.coord = new goog.math.Coordinate(x,y);
 35     /**
 36      * Bonds belonging to this atom
 37      * @type{goog.structs.Set}
 38      */
 39     this.bonds=new goog.structs.Set();
 40     /**
 41      * charge
 42      * @type{number} 
 43      */
 44     this.charge = goog.isDef(opt_charge) ? opt_charge : 0;
 45     
 46     /**
 47      * isotope
 48      * @type{number} 
 49      */
 50     this.isotope = goog.isDef(opt_isotope) ? opt_isotope : 0;
 51     
 52     /**
 53      * aromatic
 54      * @type{bool} 
 55      */
 56     this.aromatic = goog.isDef(opt_aromatic) ? opt_aromatic : false;
 57 
 58     this.hybridization=null;
 59 };
 60 
 61 jchemhub.model.Atom.prototype.countBonds = function() {
 62 	return this.bonds.getCount();	
 63 };
 64 /**
 65  * Implict hydrogen count
 66  * @return Integer
 67  */
 68 jchemhub.model.Atom.prototype.hydrogenCount = function() {
 69 	var cov = jchemhub.resource.Covalence[this.symbol];
 70 	var totalBondOrder = goog.array.reduce(this.bonds.getValues(), function(r, v) {
 71 		return r + v.constructor.ORDER;
 72 		}, 0);
 73 	var hydrogenCount = 0;
 74 	if (cov) {
 75 		hydrogenCount = cov - totalBondOrder + this.charge;
 76 	}
 77 	return hydrogenCount;
 78 };
