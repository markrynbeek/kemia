198 /**
199  * Calls all the plugins of the given operation, in sequence, with the given
200  * arguments. This is short-circuiting: once one plugin cancels the event, no
201  * more plugins will be invoked.
202  * 
203  * @param {jchemhub.controller.Plugin.Op}
204  *            op A plugin op.
205  * @param {...*}
206  *            var_args The arguments to the plugin.
207  * @return {boolean} True if one of the plugins cancel the event, false
208  *         otherwise.
209  * @private
210  */
211 jchemhub.controller.ReactionEditor.prototype.invokeShortCircuitingOp_ = function(op,
212 		var_args) {
213 	var plugins = this.indexedPlugins_[op];
214 	var argList = goog.array.slice(arguments, 1);
215 	for ( var i = 0; i < plugins.length; ++i) {
216 		// If the plugin returns true, that means it handled the event and
217 		// we shouldn't propagate to the other plugins.
218 		var plugin = plugins[i];
219 		if ((plugin.isEnabled(this) || jchemhub.controller.Plugin.IRREPRESSIBLE_OPS[op])
220 				&& plugin[jchemhub.controller.Plugin.OPCODE[op]].apply(plugin,
221 						argList)) {
222 			// Only one plugin is allowed to handle the event. If for some
223 			// reason
224 			// a plugin wants to handle it and still allow other plugins to
225 			// handle
226 			// it, it shouldn't return true.
227 			return true;
228 		}
229 	}
230 
231 	return false;
232 };

234 /**
235  * Handles keyboard shortcuts on the editor. Note that we bake this into our
236  * handleKeyPress/handleKeyDown rather than using goog.events.KeyHandler or
237  * goog.ui.KeyboardShortcutHandler for performance reasons. Since these are
238  * handled on every key stroke, we do not want to be going out to the event
239  * system every time.
240  * 
241  * @param {goog.events.BrowserEvent}
242  *            e The browser event.
243  * @private
244  */
245 jchemhub.controller.ReactionEditor.prototype.handleKeyboardShortcut_ = function(e) {
246 	// Alt key is used for i18n languages to enter certain characters. like
247 	// control + alt + z (used for IMEs) and control + alt + s for Polish.
248 	// So we don't invoke handleKeyboardShortcut at all for alt keys.
249 	if (e.altKey) {
250 		return;
251 	}
252 
253 	var isModifierPressed = goog.userAgent.MAC ? e.metaKey : e.ctrlKey;
254 	if (isModifierPressed
255 			|| jchemhub.controller.ReactionEditor.POTENTIAL_SHORTCUT_KEYCODES_[e.keyCode]) {
256 		// TODO: goog.events.KeyHandler uses much more complicated logic
257 		// to determine key. Consider changing to what they do.
258 		var key = e.charCode || e.keyCode;
259 
260 		if (key == 17) { // Ctrl key
261 			// In IE and Webkit pressing Ctrl key itself results in this event.
262 			return;
263 		}
264 
265 		var stringKey = String.fromCharCode(key).toLowerCase();
266 		if (this.invokeShortCircuitingOp_(jchemhub.controller.Plugin.Op.SHORTCUT, e,
267 				stringKey, isModifierPressed)) {
268 			e.preventDefault();
269 			// We don't call stopPropagation as some other handler outside of
270 			// trogedit might need it.
271 		}
272 	}
273 };

381 /**
382  * Gets the value of this command.
383  * 
384  * @param {string}
385  *            command The command to check.
386  * @param {boolean}
387  *            isEditable Whether the field is currently editable.
388  * @return {string|boolean|null} The state of this command. Null if not handled.
389  *         False if the field is uneditable and there are no handlers for
390  *         uneditable commands.
391  * @private
392  */
393 jchemhub.controller.ReactionEditor.prototype.queryCommandValueInternal_ = function(
394 		command, isEditable) {
395 	var plugins = this.indexedPlugins_[jchemhub.controller.Plugin.Op.QUERY_COMMAND];
396 	for ( var i = 0; i < plugins.length; ++i) {
397 		var plugin = plugins[i];
398 		if (plugin.isEnabled(this) && plugin.isSupportedCommand(command)
399 				&& (isEditable || plugin.activeOnUneditableEditors())) {
400 			return plugin.queryCommandValue(command);
401 		}
402 	}
403 	return isEditable ? null : false;
404 };
405 
406 /**
407  * Gets the value of command(s).
408  * 
409  * @param {string|Array.<string>} commands String name(s) of the command.
410  * @return {*} Value of each command. Returns false (or array of falses) if
411  *         designMode is off or the editor is otherwise uneditable, and there
412  *         are no activeOnUneditable plugins for the command.
413  */
414 jchemhub.controller.ReactionEditor.prototype.queryCommandValue = function(commands) {
415 	var isEditable = this.isLoaded();
416 	if (goog.isString(commands)) {
417 		return this.queryCommandValueInternal_(commands, isEditable);
418 	} else {
419 		var state = {};
420 		for ( var i = 0; i < commands.length; i++) {
421 			state[commands[i]] = this.queryCommandValueInternal_(commands[i],
422 					isEditable);
423 		}
424 		return state;
425 	}
426 };

502 /**
503  * Executes an editing command as per the registered plugins.
504  * 
505  * @param {string}
506  *            command The command to execute.
507  * @param {...*}
508  *            var_args Any additional parameters needed to execute the command.
509  * @return {Object|boolean} False if the command wasn't handled, otherwise, the
510  *         result of the command.
511  */
512 jchemhub.controller.ReactionEditor.prototype.execCommand = function(command, var_args) {
513 	var args = arguments;
514 	var result;
515 
516 	var plugins = this.indexedPlugins_[jchemhub.controller.Plugin.Op.EXEC_COMMAND];
517 	for ( var i = 0; i < plugins.length; ++i) {
518 		// If the plugin supports the command, that means it handled the
519 		// event and we shouldn't propagate to the other plugins.
520 		var plugin = plugins[i];
521 		if (plugin.isEnabled(this) && plugin.isSupportedCommand(command)) {
522 			result = plugin.execCommand.apply(plugin, args);
523 			break;
524 		}
525 	}
526 
527 	return result;
528 };

530 /**
531  * Registers the plugin with the editor.
532  * 
533  * @param {jchemhub.controller.Plugin}
534  *            plugin The plugin to register.
535  */
536 jchemhub.controller.ReactionEditor.prototype.registerPlugin = function(plugin) {
537 	var classId = plugin.getTrogClassId();
538 
539 	if (this.plugins_[classId]) {
540 		this.logger
541 				.severe('Cannot register the same class of plugin twice [' + classId + ']');
542 	}
543 	this.plugins_[classId] = plugin;
544 
545 	// Only key events and execute should have these has* functions with a
546 	// custom
547 	// handler array since they need to be very careful about performance.
548 	// The rest of the plugin hooks should be event-based.
549 	for ( var op in jchemhub.controller.Plugin.OPCODE) {
550 		var opcode = jchemhub.controller.Plugin.OPCODE[op];
551 		if (plugin[opcode]) {
552 			this.indexedPlugins_[op].push(plugin);
553 		}
554 	}
555 	plugin.registerEditorObject(this);
556 
557 	// By default we enable all plugins for fields that are currently loaded.
558 	if (this.isLoaded()) {
559 		plugin.enable(this);
560 	}
561 };
562 
563 /**
564  * Unregisters the plugin with this editor.
565  * 
566  * @param {jchemhub.controller.Plugin}
567  *            plugin The plugin to unregister.
568  */
569 jchemhub.controller.ReactionEditor.prototype.unregisterPlugin = function(plugin) {
570 	var classId = plugin.getTrogClassId();
571 	if (!this.plugins_[classId]) {
572 		this.logger
573 				.severe('Cannot unregister a plugin that isn\'t registered.');
574 	}
575 	delete this.plugins_[classId];
576 
577 	for ( var op in jchemhub.controller.Plugin.OPCODE) {
578 		var opcode = jchemhub.controller.Plugin.OPCODE[op];
579 		if (plugin[opcode]) {
580 			goog.array.remove(this.indexedPlugins_[op], plugin);
581 		}
582 	}
583 
584 	plugin.unregisterEditorObject(this);
585 };

677 jchemhub.controller.ReactionEditor.prototype.disposeInternal = function() {
678 	if (this.isLoading() || this.isLoaded()) {
679 		this.logger.warning('Disposing an editor that is in use.');
680 	}
681 
682 	if (this.getOriginalElement()) {
683 		this.execCommand(jchemhub.controller.Command.CLEAR);
684 	}
685 
686 	this.tearDownEditorObject_();
687 	this.clearListeners_();
688 	this.originalDomHelper = null;
689 
690 	if (this.eventRegister) {
691 		this.eventRegister.dispose();
692 		this.eventRegister = null;
693 	}
694 
695 	this.removeAllWrappers();
696 
697 	if (jchemhub.controller.ReactionEditor.getActiveEditorId() == this.id) {
698 		jchemhub.controller.ReactionEditor.setActiveEditorId(null);
699 	}
700 
701 	for ( var classId in this.plugins_) {
702 		var plugin = this.plugins_[classId];
703 		if (plugin.isAutoDispose()) {
704 			plugin.dispose();
705 		}
706 	}
707 	delete (this.plugins_);
708 
709 	jchemhub.controller.ReactionEditor.superClass_.disposeInternal.call(this);
710 };

863 /**
864  * Registers a keyboard event listener on the editor. This is necessary for
865  * Gecko since the fields are contained in an iFrame and there is no way to
866  * auto-propagate key events up to the main window.
867  * 
868  * @param {string|Array.<string>} type Event type to listen for or array of event types,
869  *            for example goog.events.EventType.KEYDOWN.
870  * @param {Function}
871  *            listener Function to be used as the listener.
872  * @param {boolean=}
873  *            opt_capture Whether to use capture phase (optional, defaults to
874  *            false).
875  * @param {Object=}
876  *            opt_handler Object in whose scope to call the listener.
877  */
878 jchemhub.controller.ReactionEditor.prototype.addListener = function(type, listener,
879 		opt_capture, opt_handler) {
880 	var elem = this.getOriginalElement();
881 
882 	// On Gecko, keyboard events only reliably fire on the document element.
883 	if (elem && goog.editor.BrowserFeature.USE_DOCUMENT_FOR_KEY_EVENTS) {
884 		elem = elem.ownerDocument;
885 	}
886 	this.eventRegister.listen(elem, type, listener, opt_capture, opt_handler);
887 };

889 /**
890  * Initialize listeners on the editor.
891  * 
892  * @private
893  */
894 jchemhub.controller.ReactionEditor.prototype.setupChangeListeners_ = function() {
895 
896 	this.addListener(goog.events.EventType.BLUR, this.dispatchBlur,
897 			goog.editor.BrowserFeature.USE_MUTATION_EVENTS);
898 
899 	if (goog.editor.BrowserFeature.USE_MUTATION_EVENTS) {
900 		// Ways to detect changes in Mozilla:
901 		//
902 		// keypress - check event.charCode (only typable characters has a
903 		// charCode), but also keyboard commands lile Ctrl+C will
904 		// return a charCode.
905 		// dragdrop - fires when the user drops something. This does not
906 		// necessary
907 		// lead to a change but we cannot detect if it will or not
908 		//
909 		// Known Issues: We cannot detect cut and paste using menus
910 		// We cannot detect when someone moves something out of the
911 		// field using drag and drop.
912 		//
913 		this.setupMutationEventHandlersGecko();
914 	} else {
915 		// Ways to detect that a change is about to happen in other browsers.
916 		// (IE and Safari have these events. Opera appears to work, but we
917 		// haven't
918 		// researched it.)
919 		//
920 		// onbeforepaste
921 		// onbeforecut
922 		// ondrop - happens when the user drops something on the editable text
923 		// field the value at this time does not contain the dropped text
924 		// ondragleave - when the user drags something from the current
925 		// document.
926 		// This might not cause a change if the action was copy
927 		// instead of move
928 		// onkeypress - IE only fires keypress events if the key will generate
929 		// output. It will not trigger for delete and backspace
930 		// onkeydown - For delete and backspace
931 		//
932 		// known issues: IE triggers beforepaste just by opening the edit menu
933 		// delete at the end should not cause beforechange
934 		// backspace at the beginning should not cause beforechange
935 		// see above in ondragleave
936 		// TODO: Why don't we dispatchBeforeChange from the
937 		// handleDrop event for all browsers?
938 		this.addListener( [ 'beforecut', 'beforepaste', 'drop', 'dragend' ],
939 				this.dispatchBeforeChange);
940 		this.addListener( [ 'cut', 'paste' ], this.dispatchChange);
941 		this.addListener('drop', this.handleDrop_);
942 	}
943 
944 	// TODO: Figure out why we use dragend vs dragdrop and
945 	// document this better.
946 	var dropEventName = goog.userAgent.WEBKIT ? 'dragend' : 'dragdrop';
947 	this.addListener(dropEventName, this.handleDrop_);
948 
949 	this.addListener(goog.events.EventType.KEYDOWN, this.handleKeyDown_);
950 	this.addListener(goog.events.EventType.KEYPRESS, this.handleKeyPress_);
951 	this.addListener(goog.events.EventType.KEYUP, this.handleKeyUp_);
952 
953 	var selectionChange = goog.bind(this.dispatchSelectionChangeEvent, this);
954 	this.selectionChangeTimer_ = new goog.async.Delay(selectionChange,
955 			jchemhub.controller.ReactionEditor.SELECTION_CHANGE_FREQUENCY_);
956 
957 	this.addListener(goog.events.EventType.MOUSEDOWN, this.handleMouseDown_);
958 	this.addListener(goog.events.EventType.MOUSEUP, this.handleMouseUp_);
959 
960 	this.addEventListener(jchemhub.controller.AtomController.EventType.MOUSEOUT, this.handleAtomMouseOut_);
961 	this.addEventListener(jchemhub.controller.AtomController.EventType.MOUSEOVER, this.handleAtomMouseOver_);
962 	this.addEventListener(jchemhub.controller.AtomController.EventType.MOUSEDOWN, this.handleAtomMouseDown_);
963 	this.addEventListener(jchemhub.controller.BondController.EventType.MOUSEOUT, this.handleBondMouseOut_);
964 	this.addEventListener(jchemhub.controller.BondController.EventType.MOUSEOVER, this.handleBondMouseOver_);
965 	this.addEventListener(jchemhub.controller.BondController.EventType.MOUSEDOWN, this.handleBondMouseDown_);
966 
967 };

974 jchemhub.controller.ReactionEditor.prototype.setupMutationEventHandlersGecko = function() {

977 				this.handleMutationEventGecko_);

981 				jchemhub.controller.ReactionEditor.MUTATION_EVENTS_GECKO,

985 		// goes
986 		// through a different handler so that we can ignore many of these.
987 		this.eventRegister.listen(doc, 'DOMAttrModified',
988 				goog.bind(this.handleDomAttrChange, this,
989 						this.handleMutationEventGecko_), true);
990 	}
