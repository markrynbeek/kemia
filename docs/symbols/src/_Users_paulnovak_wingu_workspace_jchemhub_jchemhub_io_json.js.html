 27 /**
 28  * uses JSON.parse and .stringify; needs def for IE and ?? This allows for a
 29  * JSON external representation that uses bond atom-index instead of atom
 30  * objects. So, there are 3 types of things of import here: 1. The actual
 31  * Molecule object (typically mol here) 2. The object (typically jmol here) in
 32  * which bond's atom objects are recast as atom indexes 3. The string
 33  * representaion of jmol There is not use for the JSON string represention of
 34  * the actual Molecule object.
 35  */

 66 /**
 67  * maps bond class to bond type code
 68  * 
 69  * @param{jchemhub.model.Bond} bond
 70  * @return{jchembun.io.json.BondType}
 71  */
 72 jchemhub.io.json.getTypeCode = function(bond){
 73 	if (bond instanceof jchemhub.model.SingleBond){
 74 		return jchemhub.io.json.BondType.SINGLE_BOND;
 75 	}
 76 	if (bond instanceof jchemhub.model.DoubleBond){
 77 		return jchemhub.io.json.BondType.DOUBLE_BOND;
 78 	}
 79 	if (bond instanceof jchemhub.model.TripleBond){
 80 		return jchemhub.io.json.BondType.TRIPLE_BOND;
 81 	}
 82 	throw new Error("Invalid bond type [" + bond + "]");
 83 	
 84 };
 85 
 86 /**
 87  * maps bond class to stereo type code
 88  * 
 89  * @param{jchemhub.model.Bond} bond
 90  * @return{jchemhub.io.json.StereoType}
 91  */
 92 jchemhub.io.json.getStereoCode = function(bond){
 93 	if (bond instanceof jchemhub.model.SingleBondUp){
 94 		return jchemhub.io.json.StereoType.SINGLE_BOND_UP;
 95 	}
 96 	if (bond instanceof jchemhub.model.SingleBondDown){
 97 		return jchemhub.io.json.StereoType.SINGLE_BOND_DOWN;
 98 	}
 99 	if (bond instanceof jchemhub.model.SingleBondUpOrDown){
100 		return jchemhub.io.json.StereoType.SINGLE_BOND_UP_OR_DOWN;
101 	}
102 	return jchemhub.io.json.StereoType.NOT_STEREO;
103 	
104 	
105 }

107 /**
108  * factory method for bonds
109  * 
110  * @param{jchemhub.io.json.BondType}type bond-type code
111  * @param{jchemhub.io.json.StereoType}stereo stereo-type code
112  * @param{jchemhub.model.Atom} source atom at source end of bond
113  * @param{jchemhub.model.Atom} target atom at target end of bond
114  * 
115  * @return{jchemhub.model.Bond}
116  */
117 jchemhub.io.json.createBond = function(type, stereo, source, target) {
118 	switch (type) {
119 	case jchemhub.io.json.BondType.SINGLE_BOND:
120 		switch (stereo) {
121 		case jchemhub.io.json.StereoType.NOT_STEREO:
122 			return new jchemhub.model.SingleBond(source, target);
123 		case jchemhub.io.json.StereoType.SINGLE_BOND_UP:
124 			return new jchemhub.model.SingleBondUp(source, target);
125 		case jchemhub.io.json.StereoType.SINGLE_BOND_UP_OR_DOWN:
126 			return new jchemhub.model.SingleBondUpOrDown(source, target);
127 		case jchemhub.io.json.StereoType.SINGLE_BOND_DOWN:
128 			return new jchemhub.model.SingleBondDown(source, target);
129 		default:
130 			throw new Error("invalid bond type/stereo [" + type + "]/["
131 					+ stereo + "]");
132 		};
133 	case jchemhub.io.json.BondType.DOUBLE_BOND:
134 		return new jchemhub.model.DoubleBond(source, target);
135 	case jchemhub.io.json.BondType.TRIPLE_BOND:
136 		return new jchemhub.model.TripleBond(source, target);
137 	case jchemhub.io.json.BondType.AROMATIC:
138 		return new jchemhub.model.AromaticBond(source, target);
139 	case jchemhub.io.json.BondType.SINGLE_OR_DOUBLE:
140 	case jchemhub.io.json.BondType.SINGLE_OR_AROMATIC:
141 	case jchemhub.io.json.BondType.DOUBLE_OR_AROMATIC: 
142 	case jchemhub.io.json.BondType.ANY: 
143 	default:
144 		throw new Error("invalid bond type/stereo [" + type + "]/[" + stereo
145 				+ "]");
146 	};
147 };

150 /**
151  * convert jmol JSON object or string to molecule
152  * 
153  * @param{string} arg
154  * @return{jchemhub.model.Molecule}
155  */
156 jchemhub.io.json.readMolecule = function(arg) {
157 	var jmol;
158 	if (arg.constructor == String) {
159 		jmol = goog.json.parse(arg);
160 	} else {
161 		jmol = arg;
162 	}
163 	var mol = new jchemhub.model.Molecule();
164 	mol.name = jmol.name;
165 	goog.array.forEach(jmol.atoms, function(a){
166 		mol.addAtom(new jchemhub.model.Atom(a.symbol, a.coord.x, a.coord.y, a.charge));
167 	});
168 	goog.array.forEach(jmol.bondindex, function(b){
169 		mol.addBond(jchemhub.io.json.createBond(b.type, b.stereo, mol.getAtom(b.source), mol.getAtom(b.target)));
170 	});
171 	
172 	return mol;
173 };

175 jchemhub.io.json.writeMolecule = function(mol) {
176 	return new goog.json.Serializer().serialize(jchemhub.io.json.moleculeToJson(mol));
177 };

180 /**
181  * convert molecule object to json representation
182  * 
183  * @param{jchemhub.model.Molecule} mol the molecule to convert
184  * @returns{object} in json molecule format
185  */
186 jchemhub.io.json.moleculeToJson = function(mol) {
187 	var atoms = goog.array.map(mol.atoms, function(a){
188 		return {symbol: a.symbol, coord:{x: a.coord.x, y: a.coord.y}, charge: a.charge};
189 	});
190 	var bonds = goog.array.map(mol.bonds, function(b){
191 		var btype =   jchemhub.io.json.getTypeCode(b);
192 		var bstereo = jchemhub.io.json.getStereoCode(b);
193 		return { source : mol.indexOfAtom(b.source), target : mol.indexOfAtom(b.target), type : btype, stereo : bstereo
194 		}
195 	});
196 
197 	return {
198 		name : mol.name,
199 		atoms : atoms,
200 		bondindex : bonds
201 	};
202 };

204 /**
205  * convert JSON reaction representation to reaction object
206  * 
207  * @param {string|Object}
208  *            arg The JSON object string, or object itself
209  * @return {jchemhub.model.Reaction}
210  */
211 jchemhub.io.json.readReaction = function(arg) {
212 	var jrxn;
213 	if (arg.constructor == String) {
214 		jrxn = goog.json.parse(arg);
215 	} else {
216 		jrxn = arg;
217 	}
218 	var rxn = new jchemhub.model.Reaction();
219 	rxn.header = jrxn.header;
220 	rxn.reactants = goog.array.map(jrxn.reactants, jchemhub.io.json.readMolecule);
221 	rxn.products = goog.array.map(jrxn.products, jchemhub.io.json.readMolecule);
222 	return rxn;
223 };
224 
225 /**
226  * converts a reaction object to JSON representation
227  * 
228  * @param{jchemhub.model.Reaction} rxn. The reaction to convert to json
229  * @return{object} json representation
230  */
231 jchemhub.io.json.reactionToJson = function (rxn) {
232 	var header = rxn.header;
233 	var reactants = goog.array.map(rxn.reactants, jchemhub.io.json.moleculeToJson);
234 	var products = goog.array.map(rxn.products, jchemhub.io.json.moleculeToJson);
235 	return {header: header,
236 		reactants: reactants,
237 		products: products};
238 };

240 jchemhub.io.json.writeReaction = function(rxn){
241 	return new goog.json.Serializer().serialize(jchemhub.io.json.reactionToJson(rxn));
242 }
