 28     /**
 29      * <h2 class="fixedFont">Introduction</h2>
 30      * <p>Substructure search is essentially searching for a mapping of a query on a 
 31      * molecule. In jchemhub, a flexible set of classes are used for substructure 
 32      * searching. These classes can be devied in three groups: queries, query compilers
 33      * and mappers.</p>
 34      * <h2 class="fixedFont">Queries</h2>
 35      * <p>Queries define the (sub)stucture to be searched for. Similar to a molecule, a
 36      * query contains atoms and bonds. The methods required for queries and their atoms
 37      * and bonds are defined by interfaces. These are {@link jchemhub.query.IQuery},
 38      * {@link jchemhub.query.IQueryAtom} and {@link jchemhub.query.IQueryBond}. Default
 39      * implementations for these classes are provided but any objects implementing the
 40      * interfaces can be used. The default implementations are {@link jchemhub.query.Query},
 41      * {@link jchemhub.query.QueryAtom} and {@link jchemhub.query.QueryBond}.</p> 
 42      * <p>When doing a substructure search, queries are usually created using a query
 43      * compiler. Directly manipulating query objects is only needed when implementing 
 44      * new query compilers.</p>
 45      * <h2 class="fixedFont">Query Compilers</h2>
 46      * <p>Query Compilers convert various query representations to query objects 
 47      * implementing the query interfaces. The query compiler interface 
 48      * {@link jchemhub.query.IQueryCompiler} only contains a static compile method.</p>
 49      * <p>The {@link jchemhub.query.MoleculeCompiler} takes a molecule and converts it
 50      * to a query. The resulting query is an exact (sub)structure search. Another 
 51      * smililar query compiler is the {@link jchemhub.query.SmilesCompiler} which 
 52      * converts a smiles string to a query. Complex queries can be created using the 
 53      * {@link jchemhub.query.SmartCompiler}. Smarts strings are complex queries with 
 54      * logical operators for query atoms and/or query bonds.</p>
 55      *
 56      * The usage of various query compilers is the same:
 57      * <pre class="code">
 58      * goog.require('jchemhub.query.MoleculeCompiler')
 59      * goog.require('jchemhub.query.SmilesCompiler')
 60      * goog.require('jchemhub.query.SmartsCompiler')
 61      * ...
 62      * var query = jchemhub.query.MoleculeCompiler.compile(molecule);
 63      * var query = jchemhub.query.SmilesCompiler.compile("OCCC=N");
 64      * var query = jchemhub.query.SmartsCompiler.compile("[C,N]=&@C");
 65      * </pre>
 66      * 
 67      * <h2 class="fixedFont">Mappers</h2>
 68      * <p>Mappers search for a mapping of the query onto a molecule. The interface 
 69      * for mappers is {@link jchemhub.query.IMapper}. The query is specified when
 70      * constructing the mapper object. Once constructed, one or more molecules can
 71      * be searched with the same query. Every mapper implements the mapFirst and 
 72      * mapUnique methods to map only the first found match or map all unique 
 73      * mappings.</p>
 74      * <p>Mapping all unique mappings can be time consuming for large molecules. To
 75      * solv this problem, the work of a substructure search can be divided in short
 76      * running time parts. To make use of this, the mapUniqueCallback method can be
 77      * used. Depending on the size of your molecule mapUnique will return within 
 78      * 100ms. When searching a large number of molecules (e.g. a database) it is
 79      * recommended to frequently allow the browser to do gui updating. If this is 
 80      * not done, the browser will "lock" and the user experiences the page as 
 81      * unresponsive.</p> 
 82      *
 83      * Example showing how to allow the browser to remain responsive:
 84      * <pre class="code">
 85      * var i = 0;
 86      * var smiles = ['CC(C)C', 'c1ccccc1CC', ...];
 87      * var query = jchemhub.query.SmilesCompiler.compile('c1ccccc1');
 88      * var mapper = new jchemhub.query.DFSMapper(query);
 89      * function doNext() {
 90      *       var molecule = jchemhub.io.smiles.parse(smiles[i]);
 91      *       var maps = mapper.mapUnique(molecule);
 92      *       debug("match " + (i+1) + ": " + maps.length);
 93      *       i++;
 94      *       if (i < smiles.length) {
 95      *           // a timeout of 0 means: process events (i.e. gui, mouse presses,
 96      *           // key presses, ...) and class doNext
 97      *           setTimeout(doNext, 0);
 98      *       }
 99      * }
100      * </pre>
101      *
102      *
103      * <h2 class="fixedFont">Example</h2>
104      * <pre class="code">
105      * <html>
106      *     <head>
107      *         <script>
108      *             goog.require('jchemhub.query.SmartsCompiler')
109      *             goog.require('jchemhub.query.DFSMapper')
110      *   
111      *             function doSubstructureSearch(molecule, smart) {
112      *                  var query = jchemhub.query.SmartCompiler.compile(smart);
113      *                  var mapper = jchemhub.query.DFSMapper(query);
114      *                  var maps = mapper.mapUnique(molecule);
115      *                  ...do something with found mappings...
116      *             }
117      *         </script>
118      *     </head>
119      *     <body>
120      *     </body>
121      * </html>
122      * </pre>
123      *
124      *
125      * @page Substructure Search
126      * @name needed to make sure jsdoc doesn't ignore the comment
127      */

129     /**
130      * <p>The {@link jchemhub.query.IQuery} class defines an interface for Query
131      * objects. Other classes in the Substructure Search framework rely on these interfaces
132      * to ensure interoperability. The interface doesn't actually do anything. Since 
133      * javascript is a dynamic language, we can pass any object as parameter. As long
134      * as the object has the correct methods and fields, no inheritance is needed.</p>
135      * 
136      * <p>This interface makes use of two other interfaces: {@link jchemhub.query.IQueryAtom} 
137      * and {@link jchemhub.query.IQueryBond}. There is a default implementation for these
138      * interfaces: {@link jchemhub.query.Query}, {@link jchemhub.query.QueryAtom} and
139      * {@link jchemhub.query.QueryBond}. However, in practice the use of query compilers 
140      * limit the need to directly manipulate query objects. There are query compilers to 
141      * convert a {@link jchemhub.model.Molecule}, SMILES or SMARTS to a query.</p>
142      * 
143      *
144      * Example showing the use of various query compilers:
145      * <pre class="code">
146      * goog.require('jchemhub.query.MoleculeCompiler')
147      * goog.require('jchemhub.query.SmilesCompiler')
148      * goog.require('jchemhub.query.SmartsCompiler')
149      * ...
150      * var query = jchemhub.query.MoleculeCompiler.compile(molecule);
151      * var query = jchemhub.query.SmilesCompiler.compile("OCCC=N");
152      * var query = jchemhub.query.SmartsCompiler.compile("[C,N]=&@C");
153      * </pre>

157      * @interface
158      */

180          */

182         /**
183          * Find the bond connecting atom1 with atom2.

316     /**
317      * Find the bond connecting atom1 with atom2.
318      * @param atom1 The first atom.
319      * @param atom2 The second atom.
320      * @return {?jchemhub.query.QueryBond} The found bond or null if no such bond exists.
321      */
322     jchemhub.query.Query.prototype.findBond = function(/**jchemhub.query.QueryAtom*/atom1, /**jchemhub.query.QueryAtom*/atom2) {
323         for ( var i = 0, il = this.bonds.length; i < il; i++) {
324             var bond = this.bonds[i];
325             if ((atom1 === bond.source && atom2 === bond.target) || (atom2 === bond.source && atom1 === bond.target)) {
326                 return bond;
327             }
328         }
329         return null;
330     };

346     /**
347      * Compare an atom with this query atom.
348      * @param atom The atom to be compared with this query atom.
349      * @param opt_molecule The molecule containing atom.
350      * @param opt_sssr The SSSR for the molecule.
351      * @return {boolean} True if the specified atom matches this query atom.
352      */
353     jchemhub.query.QueryAtom.prototype.matches = function(/**jchemhub.query.QueryAtom*/atom, /**jchemhub.model.Moleculei=*/opt_molecule, 
354             /**Array.<jchemhub.ring.Ring>=*/opt_sssr) {
355         var symbolMatches = true;
356         if (this.symbols.length) {
357             if (goog.array.indexOf(this.symbols, atom.symbol) === -1) {
358                 symbolMatches = false;                
359             }
360         }
361 
362         var valenceMatches = this.valence <= atom.countBonds();
363 
364         return symbolMatches && valenceMatches;    
365     };
