 75 /**
 76  * Find the bond between two given atoms if it exists. Otherwise return null.
 77  * 
 78  * @param {Object}
 79  *            atom1
 80  * @param {Object}
 81  *            atom2
 82  * @return{jchemhub.model.Bond}
 83  */
 84 jchemhub.model.Molecule.prototype.findBond = function(atom1, atom2) {
 85 	for ( var i = 0, il = this.bonds.length; i < il; i++) {
 86 		var bond = this.bonds[i];
 87 		if ((atom1 == bond.source && atom2 == bond.target)
 88 				|| (atom2 == bond.source && atom1 == bond.target))
 89 			return bond;
 90 	}
 91 	return null;
 92 };

116 /**
117  * Remove a atom from molecule.
118  * 
119  * @param {number|jchemhub.model.Atom}
120  *            Instance or id of the atom to remove.
121  */
122 
123 jchemhub.model.Molecule.prototype.removeAtom = function(atomOrId) {
124 	var atom;
125 	if (atomOrId.constructor == Number) {
126 		atom = this.atoms[atomOrId];
127 	} else if (atomOrId.constructor == jchemhub.model.Atom) {
128 		atom = atomOrId;
129 	}
130 	var neighborBonds = atom.bonds.getValues();
131 	var molBonds = this.bonds; // for bond reference in anonymous method
132 	goog.array.forEach(neighborBonds, function(element, index, array) {
133 		goog.array.remove(molBonds, element);
134 	});
135 	atom.bonds.clear();
136 	goog.array.remove(this.atoms, atom);
137 	atom.molecule = undefined;
138 
139 };
