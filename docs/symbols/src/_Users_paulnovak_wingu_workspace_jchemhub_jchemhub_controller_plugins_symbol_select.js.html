 52 jchemhub.controller.plugins.SymbolSelect.prototype.handleAtomMouseDown = function(
 53 		e) {
 54 	if (this.symbol && (this.symbol!=e.atom.symbol)) {
 55 		var new_atom = new jchemhub.model.Atom(this.symbol, e.atom.coord.x, e.atom.coord.y)
 56 		goog.array.forEach(e.atom.bonds.getValues(), function(bond){
 57 			var new_bond = bond.clone();
 58 			new_bond.molecule = undefined;
 59 			e.atom==new_bond.source ? new_bond.source = new_atom : new_bond.target = new_atom;
 60 			e.atom.molecule.addBond(new_bond);
 61 		});
 62 		var molecule = e.atom.molecule
 63 		molecule.removeAtom(e.atom);
 64 		molecule.addAtom(new_atom);
 65 		this.editorObject.setModel(this.editorObject.getModel());
 66 	}
 67 };
