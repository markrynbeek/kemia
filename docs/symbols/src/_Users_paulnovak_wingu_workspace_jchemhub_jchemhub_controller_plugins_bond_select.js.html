 86 jchemhub.controller.plugins.BondSelect.prototype.handleBondMouseDown = function(
 87 		e) {
 88 
 89 	if (this.bond_klass){
 90 		var new_bond = new this.bond_klass(e.bond.source, e.bond.target);
 91 		var molecule = e.bond.molecule;
 92 		molecule.removeBond(e.bond);
 93 		molecule.addBond(new_bond);
 94 		this.editorObject.setModel(this.editorObject.getModel());
 95 	}
 96 
 97 };
