 38 /**
 39  * Hanser main loop, produces the rings for a given molecule.
 40  * @param {Object} _molecule
 41  */
 42 jchemhub.ring.RingFinder.findRings = function(_molecule){
 43 
 44     var molecule = _molecule;
 45     var atomOnlyRings = new Array();
 46 
 47     graph = new jchemhub.ring.PathGraph(molecule)
 48 
 49     for (var i = 0,il = molecule.countAtoms(); i < il; i++) {
 50         var edges = graph.remove(molecule.getAtom(i));
 51         for (var j = 0; j < edges.length; j++) {
 52             edge = edges[j];
 53             atom_ring = edge.atoms;
 54             atomOnlyRings.push(atom_ring);
 55         }
 56     }
 57     //xtra: sort array according to ring size
 58     goog.array.sort(atomOnlyRings);
 59 	
 60 	var rings=new Array();
 61     for (var i = 0, il = atomOnlyRings.length; i < il; i++) {
 62 		rings.push(this.createRing(atomOnlyRings[i],molecule));
 63 	}
 64     return rings;
 65 }

 67 /**
 68  * The Hanser Ring Finder produces a ring as just a series of atoms.
 69  * Here we complete this information with the bonds and the ring center,
 70  * creating a ring object.
 71  * @param {Object} atoms
 72  */
 73 jchemhub.ring.RingFinder.createRing = function(atoms,molecule){
 74 	
 75     var bonds = new Array();
 76     for (var i = 0, il = atoms.length-1; i < il; i++) {
 77 		bond = molecule.findBond(atoms[i],atoms[i+1]);
 78 		if(bond!=null) {
 79 			bonds.push(bond);
 80 		}
 81 	}
 82     //Hanser last atom is same as first atom, remove it..
 83     goog.array.removeAt(atoms, atoms.length - 1);
 84     
 85 	var ring = new jchemhub.ring.Ring(atoms,bonds);  
 86     return ring;
 87 }

 94 jchemhub.ring.PathGraph = function(molecule){
 95     this.edges = new Array();
 96     this.atoms = new Array();
 97     // load edges
 98     for (var i = 0,il = molecule.countBonds(); i < il; i++) {
 99         bond = molecule.getBond(i);
100         var edge = [bond.source, bond.target];
101         this.edges.push(new jchemhub.ring.PathEdge(edge));
102     }
103     // load atoms
104     for (var i = 0,il=molecule.countAtoms(); i < il; i++) {
105         this.atoms.push(molecule.getAtom(i));
106     }
107 }
108 
109 jchemhub.ring.PathGraph.prototype.remove = function(atom){
110     var oldEdges = this.getEdges(atom);
111     result = new Array();
112     for (var i = 0, il = oldEdges.length; i < il; i++) {
113         if (oldEdges[i].isCycle()) {
114             result.push(oldEdges[i]);
115         }
116     }
117 
118     for (var i = 0,il = result.length; i < il; i++) {
119         if (goog.array.contains(oldEdges, result[i])) {
120             goog.array.remove(oldEdges, result[i]);
121         }
122         if (goog.array.contains(this.edges, result[i])) {
123             goog.array.remove(this.edges, result[i]);
124         }
125     }
126     
127     newEdges = this.spliceEdges(oldEdges);
128     
129     for (var i = 0,il=oldEdges.length; i < il; i++) {
130         if (goog.array.contains(this.edges, oldEdges[i])) {
131             goog.array.remove(this.edges, oldEdges[i]);
132         }
133     }
134     
135     for (var i = 0,il=newEdges.length; i < il; i++) {
136         if (!goog.array.contains(this.edges, newEdges[i])) {
137             this.edges.push(newEdges[i]);
138         }
139     }
140     goog.array.remove(this.atoms, atom);
141     return result;
142 }

145 jchemhub.ring.PathGraph.prototype.getEdges = function(atom){
146     var result = new Array();
147     
148     for (var i = 0,il=this.edges.length; i < il; i++) {
149         edge = this.edges[i];
150         
151         if (edge.isCycle()) {
152             if (goog.array.contains(edge.atoms, atom)) {
153                 result.push(edge);
154             }
155         }
156         else {
157             var lastAtomPos = edge.atoms.length - 1;
158             if ((edge.atoms[0] == atom) || (edge.atoms[lastAtomPos] == atom)) {
159                 result.push(edge);
160             }
161         }
162     }
163     return result;
164 }
165 
166 jchemhub.ring.PathGraph.prototype.spliceEdges = function(_edges){
167     var result = new Array();
168     
169     for (var i = 0,il=_edges.length; i < il; i++) {
170         for (var j = i + 1; j < il; j++) {
171             spliced = _edges[j].splice(_edges[i]);
172             if (spliced != null) {
173                 result.push(spliced);
174             }
175         }
176     }
177     return result;
178 }

187     this.atoms = _atoms;

190 jchemhub.ring.PathEdge.prototype.isCycle = function(){
191     var lastAtomPos = this.atoms.length - 1;
192     return (this.atoms.length > 2 && this.atoms[0] == this.atoms[lastAtomPos]);
193 }

196 jchemhub.ring.PathEdge.prototype.splice = function(other){
197     intersection = this.getIntersection(other.atoms);
198     newAtoms = new Array();
199     for (var i = 0,il=this.atoms.length; i < il; i++) {
200         newAtoms.push(this.atoms[i]);
201     }
202     
203     if (this.atoms[0] == intersection) {
204         newAtoms.reverse();
205     }
206     
207     if (other.atoms[0] == intersection) {
208         for (var i = 1,il=other.atoms.length; i < il; i++) {
209             newAtoms.push(other.atoms[i]);
210         }
211     }
212     else {
213         for (var i = other.atoms.length - 2; i >= 0; i--) {
214             newAtoms.push(other.atoms[i]);
215         }
216     }
217     
218     if (!this.isRealPath(newAtoms)) {
219         return null;
220     }
221     
222     return new jchemhub.ring.PathEdge(newAtoms);
223 }
