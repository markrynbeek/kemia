 27 /*
 28  * Return a bond of the correct subclass given the input arguments.
 29  * TODO: extend with stereo types.
 30  */
 31 jchemhub.util.BondUtil.getBond = function(atom1, atom2, bondOrder){
 32     bond=null;
 33     switch (bondOrder) {
 34         case jchemhub.util.BondUtil.Orders.SINGLE:
 35             bond = new jchemhub.model.SingleBond(atom1,atom2);
 36             break;
 37         case jchemhub.util.BondUtil.Orders.DOUBLE:
 38             bond = new jchemhub.model.DoubleBond(atom1,atom2);
 39             break;
 40         case jchemhub.util.BondUtil.Orders.TRIPLE:
 41             bond = new jchemhub.model.TripleBond(atom1,atom2);
 42             break;
 43         case jchemhub.util.BondUtil.Orders.QUADRUPLE:
 44             bond = new jchemhub.model.QuadrupleBond(atom1,atom2);
 45             break;
 46     }
 47     return bond;
 48 }

 55 jchemhub.util.BondUtil.Orders = {

 57         DOUBLE :2,
 58         TRIPLE :3,
 59         QUADRUPLE :4
