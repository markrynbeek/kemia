 43 /**
 44  * Parse a SMILES String into a jchemhub.model.Molecule.
 45  */
 46 jchemhub.io.smiles.SmilesParser.parse = function(smiles){
 47 
 48     //Initialization before parse 
 49     var bond = new jchemhub.model.Bond;
 50     var nodeCounter = 0;
 51     var bondOrder = null;  
 52     var bondOrderForRingClosure = jchemhub.util.BondUtil.Orders.SINGLE;
 53     var bondIsAromatic = false; 
 54     var bondExists = true;
 55     var thisRing = -1;
 56     var currentSymbol = null;
 57     var molecule=new jchemhub.model.Molecule;
 58     var position = 0;
 59     var rings = new Array();  // array with atoms
 60     var ringbonds = new Array(); // array with bonds 
 61     var mychar = 'X';
 62     var chars = new Array(); // char[1];
 63     var lastNode = null; //Atom
 64     var atomStack = new Array(); // stack with atoms
 65     var bondStack = new Array(); // stack with bond orders
 66     var atom = null;
 67 
 68     //Parser loep
 69 	do {
 70         mychar = smiles.charAt(position);
 71 
 72         if ((mychar >= 'A' && mychar <= 'Z') || (mychar >= 'a' && mychar <= 'z') ||(mychar == '*'))
 73         {
 74             status = 1;
 75             atom = null;
 76 			// A star means a pseudo atom 
 77             if (mychar == '*') {
 78                 currentSymbol = "*";
 79                 atom = new jchemhub.model.PseudoAtom("*",0,0);
 80             }
 81 			else {
 82                 currentSymbol = this.getSymbolForOrganicSubsetElement(smiles,position);
 83 				//alert(currentSymbol);
 84 				if (currentSymbol != null) {
 85                     if (currentSymbol.length == 1) {
 86                         if (currentSymbol.toUpperCase() != currentSymbol ) { 
 87                             currentSymbol = currentSymbol.toUpperCase();
 88                             atom = new jchemhub.model.Atom(currentSymbol);
 89 							atom.hybridization=jchemhub.model.Atom.Hybridizations.SP2;
 90                         } 
 91 						else
 92                             atom = new jchemhub.model.Atom(currentSymbol);
 93                     } 
 94 					else
 95                         atom = new jchemhub.model.Atom(currentSymbol);
 96                 } else
 97                      throw "Found element which is not a 'organic subset' element. You must use [" + mychar + "].";
 98             }
 99 			molecule.addAtom(atom); 
100 
101             if ((lastNode != null) && bondExists) {
102 				bond = jchemhub.util.BondUtil.getBond (atom, lastNode,bondOrder);
103                 if (bondIsAromatic) {
104                     bond.aromatic=true;
105                 }
106                 molecule.addBond(bond);
107             }
108             bondOrder = jchemhub.util.BondUtil.Orders.SINGLE;
109             lastNode = atom;
110             nodeCounter++;
111             position = position + currentSymbol.length;
112             bondExists = true;
113             bondIsAromatic = false;
114         } 
115 
116         else if (mychar == '=') {
117 			//alert(mychar);
118             position++;
119             if (status == 2 || !((smiles.charAt(position) >= '0' && smiles.charAt(position) <= '9') || smiles.charAt(position) == '%'))
120                 bondOrder = jchemhub.util.BondUtil.Orders.DOUBLE;
121             else
122                 bondOrderForRingClosure = jchemhub.util.BondUtil.Orders.DOUBLE;
123         } 
124     	else if (mychar == '#') {
125             //alert(mychar);
126             position++;
127             if (status == 2 || !((smiles.charAt(position) >= '0' && smiles.charAt(position) <= '9') || smiles.charAt(position) == '%')) 
128                 bondOrder = jchemhub.util.BondUtil.Orders.TRIPLE;
129             else 
130                 bondOrderForRingClosure = jchemhub.util.BondUtil.Orders.TRIPLE;
131         } 
132 		else if (mychar == '(') {
133             //alert(mychar);
134             atomStack.push(lastNode);
135             bondStack.push(bondOrder);
136             position++;
137         }
138 		else if (mychar == ')' ) {
139             //alert(mychar);
140             lastNode = atomStack.pop();
141             bondOrder = bondStack.pop();
142             position++;
143         } 
144 		else if (mychar >= '0' && mychar <= '9') {
145             status = 2;
146             chars[0] = mychar;
147             currentSymbol = mychar; //new String(chars);
148             thisRing = currentSymbol;
149 
150             //Handle ring__________			 
151 			var bondOrd = bondOrderForRingClosure;
152 			if (ringbonds[thisRing] > bondOrd) 
153 			    bondOrd = ringbonds[thisRing];
154 			ringBond = null;
155 			ringPartner = null;
156 			ringNode = rings[thisRing];
157 			if (ringNode != null) {
158 			    ringPartner = ringNode;
159 			    ringBond = jchemhub.util.BondUtil.getBond(lastNode, ringPartner, bondOrd);
160 			    if (bondIsAromatic) {
161 			        ringBond.setFlag(CDKConstants.ISAROMATIC, true);
162 			    }
163 			    molecule.addBond(ringBond);
164 			    bondIsAromatic = false;
165 			    rings[thisRing] = null;
166 			    ringbonds[thisRing] = null;
167 			} 
168 			else {
169 			    rings[thisRing] = lastNode;
170 			    ringbonds[thisRing] = bondOrderForRingClosure;
171 			}
172 			bondOrderForRingClosure = jchemhub.util.BondUtil.Orders.SINGLE;
173             //______________________
174             position++;
175 
176         }
177 		else if (mychar == '%')
178         {
179             pos=position+1;
180 	        if (pos >= smiles.length - 1)
181 	            throw "Percent sign ring closure numbers must be two-digit.";
182 	
183 	        currentSymbol = smiles.substring(pos, pos + 2);
184 	
185 	        if (currentSymbol.charAt(0) < '0' || currentSymbol.charAt(0) > '9' || 
186 	            currentSymbol.charAt(1) < '0' || currentSymbol.charAt(1) > '9')
187 	            throw "Percent sign ring closure numbers must be two-digit.";
188 
189             thisRing = currentSymbol;
190 
191             //Handle ring_______________            
192             var bondOrd = bondOrderForRingClosure;
193             if (ringbonds[thisRing] > bondOrd) 
194                 bondOrd = ringbonds[thisRing];
195             ringBond = null;
196             ringPartner = null;
197             ringNode = rings[thisRing];
198             if (ringNode != null) {
199                 ringPartner = ringNode;
200                 ringBond = jchemhub.util.BondUtil.getBond(lastNode, ringPartner, bondOrd);
201                 if (bondIsAromatic) {
202                     ringBond.setFlag(CDKConstants.ISAROMATIC, true);
203                 }
204                 molecule.addBond(ringBond);
205                 bondIsAromatic = false;
206                 rings[thisRing] = null;
207                 ringbonds[thisRing] = null;
208             } 
209             else {
210                 rings[thisRing] = lastNode;
211                 ringbonds[thisRing] = bondOrderForRingClosure;
212             }
213             bondOrderForRingClosure = jchemhub.util.BondUtil.Orders.SINGLE;
214 			//_____________
215 
216             position += currentSymbol.length + 1;
217         } 
218 		else if (mychar == '[')  {
219             currentSymbol = this.getAtomString(smiles, position);
220             atom = this.assembleAtom(currentSymbol);
221             molecule.addAtom(atom);
222             if (lastNode != null && bondExists) {
223                 bond = jchemhub.util.BondUtil.getBond (atom, lastNode,bondOrder);
224                 if (bondIsAromatic) {
225                     bond.aromatic=true;
226                 }
227                 molecule.addBond(bond);
228             }
229             bondOrder = jchemhub.util.BondUtil.Orders.SINGLE;
230             bondIsAromatic = false;
231             lastNode = atom;
232             nodeCounter++;
233             position = position + currentSymbol.length + 2;
234 
235             //TODO change atom class
236             //if (atom.getHydrogenCount() == null) {
237                 // zero implicit hydrogens is implied when the Hx syntax is not used
238             //    atom.setHydrogenCount(0);
239             //}
240             bondExists = true;
241         }
242 		else if (mychar == '.')
243         {
244             bondExists = false;
245             position++;
246         } else if (mychar == '-')
247         {
248             bondExists = true;
249             position++;
250         } else if (mychar == ':') {
251             bondExists = true;
252             bondIsAromatic = true;
253             position++;
254         } else if (mychar == '/' || mychar == '\\')
255         {
256             position++;
257         } else if (mychar == '@')
258         {
259             if (position < smiles.length - 1 && smiles.charAt(position + 1) == '@')
260             {
261                 position++;
262             }
263             position++;
264         } else
265         {
266             throw "Invalid SMILES: found unexpected char: " + mychar;
267         }
268         
269 	}
270 	while (position < smiles.length)
271     return molecule;
272 }

275 /**
276  *  Gets the element symbol for an element in the 'organic subset' for which
277  *  brackets may be omitted.
278  *
279  */
280 jchemhub.io.smiles.SmilesParser.getSymbolForOrganicSubsetElement = function(s, pos){
281     var possibleSymbol="";
282 	if (pos < s.length - 1){
283         possibleSymbol = s.substring(pos, pos + 2);
284         if (("ClBr".indexOf(possibleSymbol) >= 0)){
285             return possibleSymbol;
286         }
287     }
288     if ("BCcNnOoFPSsI".indexOf((s.charAt(pos))) >= 0){
289         return s.substring(pos, pos + 1);
290     }
291     if ("fpi".indexOf((s.charAt(pos))) >= 0){
292         return s.substring(pos, pos + 1);
293     }
294     alert("SMILES parser problem: subset element not found at position "+pos);
295     return null;
296 }

299 jchemhub.io.smiles.SmilesParser.getAtomString = function(smiles, pos) {
300     atomString = "";
301     for (f=pos+1; f<smiles.length; f++) {
302         character = smiles.charAt(f);
303         if (character == ']')
304            break;
305         else
306            atomString+=character;
307     }
308     return atomString;
309 }

312 jchemhub.io.smiles.SmilesParser.getElementSymbol = function(s,pos) {
313     // Try to match elements not in the organic subset.
314     // first, the two char elements
315     if (pos < s.length - 1)
316     {
317         possibleSymbol = s.substring(pos, pos + 2);
318         if (("HeLiBeNeNaMgAlSiClArCaScTiCrMnFeCoNiCuZnGaGeAsSe".indexOf(possibleSymbol) >= 0) ||
319             ("BrKrRbSrZrNbMoTcRuRhPdAgCdInSnSbTeXeCsBaLuHfTaRe".indexOf(possibleSymbol) >= 0) ||
320             ("OsIrPtAuHgTlPbBiPoAtRnFrRaLrRfDbSgBhHsMtDs".indexOf(possibleSymbol) >= 0))
321             return possibleSymbol;
322     }
323 
324     // if that fails, the one char elements
325     possibleSymbol = s.substring(pos, pos + 1);
326     if (("HKUVYW".indexOf(possibleSymbol) >= 0))
327         return possibleSymbol;
328 
329     // if that failed too, then possibly a organic subset element
330     return this.getSymbolForOrganicSubsetElement(s, pos);
331 }

335 jchemhub.io.smiles.SmilesParser.getImplicitHydrogenCount = function(s,position) {
336     count = 0;  
337     if (s.charAt(position) == 'H') {
338         multiplier = "";
339         while (position < (s.length - 1) && isDigit(s.charAt(position + 1))){
340             multiplier+=s.charAt(position + 1);
341             position++;
342         }
343         if (multiplier.length > 0) {
344             count = parseInt(multiplier);
345         } 
346 		else count = -1; 
347     }
348     return count;
349 }

352 jchemhub.io.smiles.SmilesParser.assembleAtom = function(s) {
353     atom = null;
354     position = 0;
355     currentSymbol = null;
356     isotopicNumber = "";
357     mychar='';
358     do {
359         mychar = s.charAt(position);
360         if ((mychar >= 'A' && mychar <= 'Z') || (mychar >= 'a' && mychar <= 'z'))
361         {
362             currentSymbol = this.getElementSymbol(s, position);
363             if (currentSymbol == null) {
364                 throw "Expected element symbol, found null!";
365             } 
366 			else {
367                 position = position + currentSymbol.length;
368                 if (currentSymbol.length == 1){
369                     if ( currentSymbol.toUpperCase()!=currentSymbol ) {
370                         currentSymbol = currentSymbol.toUpperCase();
371                         atom = new jchemhub.model.Atom(currentSymbol);
372                         atom.hybridization=jchemhub.model.Atom.Hybridizations.SP2;
373                         //TODO here: hydrogen count
374 
375                     } else
376                         atom = new jchemhub.model.Atom(currentSymbol);
377                 } else
378                         atom = new jchemhub.model.Atom(currentSymbol);
379             }
380             break;
381 
382         } else if (mychar >= '0' && mychar <= '9'){
383             isotopicNumber+=mychar;
384             position++;
385         } else if (mychar == '*') {
386             currentSymbol = "*";
387             atom = new jchemhub.model.PseudoAtom("*",0,0);
388             position++;
389             break;
390         } else
391             throw "Invalid SMILES: found unexpected char: " + mychar;
392     } 
393 	while (position < s.length);
394 
395 
396     //TODO >>> change atom class
397     //if (isotopicNumber.toString().length() > 0)
398     //{
399     //   atom.setMassNumber(Integer.parseInt(isotopicNumber.toString()));
400     //}
401 	
402     charge = 0;
403     implicitHydrogens = 0;
404     while (position < s.length)
405     {
406         mychar = s.charAt(position);
407         if (mychar == 'H')
408         {
409             implicitHydrogens = this.getImplicitHydrogenCount(s, position);
410             position++;
411             if (implicitHydrogens >= 0) {
412                 position++;
413             }
414             //TODO >> change atom class                     
415             //if (implicitHydrogens == -1) implicitHydrogens = 1;
416             //  atom.setHydrogenCount(implicitHydrogens);
417         } else if (mychar == '+' || mychar == '-')
418         {
419             charge = getCharge(s, position);
420             position++; 
421             while (position < s.length && isDigit(s.charAt(position))) {
422                 position++;
423             }
424             //TODO >> change atom class                     
425             atom.setFormalCharge(charge);
426         } else if (mychar == '@')
427         {
428             if (position
